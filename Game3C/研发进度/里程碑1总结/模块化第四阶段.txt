#include "UIManagerComponent.h"
#include "GameFramework/Character.h"
#include "GameFramework/PlayerController.h"

UUIManagerComponent::UUIManagerComponent()
{
    PrimaryComponentTick.bCanEverTick = true;
    PrimaryComponentTick.TickInterval = 0.016f; // ~60 FPS for UI updates
}

void UUIManagerComponent::BeginPlay()
{
    Super::BeginPlay();
    UpdateOwnerReferences();
    
    if (bEnableUIDebugLogs)
    {
        DebugWidgetSetup();
    }
}

void UUIManagerComponent::TickComponent(float DeltaTime, ELevelTick TickType, FActorComponentTickFunction* ThisTickFunction)
{
    Super::TickComponent(DeltaTime, TickType, ThisTickFunction);
    
    // 如果有当前目标且使用Socket投射，持续更新位置
    if (CurrentWidgetTarget && bUseSocketProjection && LockOnWidgetInstance && LockOnWidgetInstance->IsInViewport())
    {
        UpdateSocketProjectionWidget(CurrentWidgetTarget);
    }
}

void UUIManagerComponent::ShowLockOnWidget(AActor* Target)
{
    // 从MyCharacter.cpp复制ShowLockOnWidget的完整实现
    if (!Target)
        return;

    // 首先隐藏所有现有的锁定UI
    HideAllLockOnWidgets();

    CurrentWidgetTarget = Target;

    // 根据显示模式选择相应的UI系统
    switch (CurrentDisplayMode)
    {
    case EUIDisplayMode::SocketProjection:
        if (bUseSocketProjection)
        {
            ShowSocketProjectionWidget(Target);
            break;
        }
        // Fall through to Traditional3D if socket projection is disabled
    case EUIDisplayMode::Traditional3D:
        HandleTraditional3DWidget(Target);
        break;
    case EUIDisplayMode::ScreenSpace:
        HandleScreenSpaceWidget(Target);
        break;
    }

    // 触发事件
    OnLockOnWidgetShown.Broadcast(Target);
    PreviousWidgetTarget = Target;
    
    if (bEnableUIDebugLogs)
    {
        LogWidgetState(TEXT("ShowLockOnWidget"));
    }
}

void UUIManagerComponent::ShowSocketProjectionWidget(AActor* Target)
{
    // 从MyCharacter.cpp复制ShowSocketProjectionWidget的完整实现
    if (!Target)
    {
        UE_LOG(LogTemp, Error, TEXT("ShowSocketProjectionWidget: No target specified"));
        return;
    }

    // 如果没有设置Widget类，尝试运行时查找
    if (!LockOnWidgetClass)
    {
        if (!TryFindWidgetClassAtRuntime())
        {
            UE_LOG(LogTemp, Error, TEXT("ShowSocketProjectionWidget: Could not find any valid Widget class"));
            return;
        }
    }

    if (!OwnerPlayerController)
    {
        UE_LOG(LogTemp, Error, TEXT("ShowSocketProjectionWidget: No valid PlayerController"));
        return;
    }

    // 清理现有实例
    CleanupWidgetInstance();

    // 创建新的Widget实例
    LockOnWidgetInstance = CreateWidget<UUserWidget>(OwnerPlayerController, LockOnWidgetClass);
    if (LockOnWidgetInstance)
    {
        LockOnWidgetInstance->AddToViewport();
        
        // 立即更新一次位置
        UpdateSocketProjectionWidget(Target);
        
        if (bEnableUIDebugLogs)
        {
            UE_LOG(LogTemp, Warning, TEXT("Socket projection widget created for target: %s"), *Target->GetName());
        }
    }
    else
    {
        UE_LOG(LogTemp, Error, TEXT("Failed to create UMG widget instance"));
    }
}

void UUIManagerComponent::UpdateSocketProjectionWidget(AActor* Target)
{
    // 从MyCharacter.cpp复制UpdateSocketProjectionWidget的完整实现
    if (!Target || !LockOnWidgetInstance || !LockOnWidgetInstance->IsInViewport())
        return;

    // 获取Socket世界位置
    FVector SocketWorldLocation = GetTargetSocketWorldLocation(Target);
    
    // 投射到屏幕坐标
    FVector2D ScreenPosition = ProjectSocketToScreen(SocketWorldLocation);
    
    // 检查投射是否成功
    if (ScreenPosition != FVector2D::ZeroVector)
    {
        // 调用UMG更新函数
        if (CallWidgetUpdateFunction(ScreenPosition))
        {
            // 确保Widget可见
            if (!LockOnWidgetInstance->IsVisible())
            {
                LockOnWidgetInstance->SetVisibility(ESlateVisibility::Visible);
            }
            
            // 触发事件
            OnSocketProjectionUpdated.Broadcast(Target, ScreenPosition);
            
            if (bShowSocketProjectionDebug)
            {
                LogSocketProjectionDetails(Target, SocketWorldLocation, ScreenPosition);
            }
        }
    }
    else
    {
        // 目标在屏幕外，隐藏Widget
        if (LockOnWidgetInstance->IsVisible())
        {
            LockOnWidgetInstance->SetVisibility(ESlateVisibility::Hidden);
        }
    }
}

FVector UUIManagerComponent::GetTargetSocketWorldLocation(AActor* Target) const
{
    // 从MyCharacter.cpp复制GetTargetSocketWorldLocation的完整实现
    if (!Target)
    {
        return FVector::ZeroVector;
    }

    USkeletalMeshComponent* SkeletalMesh = Target->FindComponentByClass<USkeletalMeshComponent>();
    if (SkeletalMesh && SkeletalMesh->DoesSocketExist(TargetSocketName))
    {
        FVector SocketLocation = SkeletalMesh->GetSocketLocation(TargetSocketName);
        return SocketLocation + SocketOffset;
    }
    else
    {
        return Target->GetActorLocation() + SocketOffset;
    }
}

bool UUIManagerComponent::HasValidSocket(AActor* Target) const
{
    // 从MyCharacter.cpp复制HasValidSocket的完整实现
    if (!Target)
    {
        return false;
    }

    USkeletalMeshComponent* SkeletalMesh = Target->FindComponentByClass<USkeletalMeshComponent>();
    if (SkeletalMesh)
    {
        return SkeletalMesh->DoesSocketExist(TargetSocketName);
    }

    return false;
}

FVector2D UUIManagerComponent::ProjectSocketToScreen(const FVector& SocketWorldLocation) const
{
    // 从MyCharacter.cpp复制ProjectSocketToScreen的完整实现
    if (!OwnerPlayerController)
    {
        return FVector2D::ZeroVector;
    }

    FVector2D ScreenLocation;
    bool bProjected = OwnerPlayerController->ProjectWorldLocationToScreen(SocketWorldLocation, ScreenLocation);
    
    return bProjected ? ScreenLocation : FVector2D::ZeroVector;
}

void UUIManagerComponent::HideAllLockOnWidgets()
{
    // 从MyCharacter.cpp复制HideAllLockOnWidgets的完整实现
    
    // 隐藏Socket投射Widget
    if (bUseSocketProjection)
    {
        HideSocketProjectionWidget();
    }

    // 隐藏传统3D Widget组件
    for (auto& WidgetPair : TargetWidgetComponents)
    {
        if (IsValid(WidgetPair.Value) && WidgetPair.Value->IsVisible())
        {
            WidgetPair.Value->SetVisibility(false);
        }
    }

    // 清理屏幕空间Widget
    CleanupWidgetInstance();
    
    CurrentWidgetTarget = nullptr;
}

void UUIManagerComponent::DebugWidgetSetup()
{
    // 从MyCharacter.cpp复制DebugWidgetSetup的完整实现
    UE_LOG(LogTemp, Warning, TEXT("=== UI MANAGER WIDGET SETUP DEBUG ==="));
    UE_LOG(LogTemp, Warning, TEXT("LockOnWidgetClass is set: %s"), LockOnWidgetClass ? TEXT("YES") : TEXT("NO"));
    
    if (LockOnWidgetClass)
    {
        UE_LOG(LogTemp, Warning, TEXT("LockOnWidgetClass name: %s"), *LockOnWidgetClass->GetName());
        UE_LOG(LogTemp, Warning, TEXT("LockOnWidgetClass path: %s"), *LockOnWidgetClass->GetPathName());
    }
    
    UE_LOG(LogTemp, Warning, TEXT("Current display mode: %d"), static_cast<int32>(CurrentDisplayMode));
    UE_LOG(LogTemp, Warning, TEXT("Socket projection enabled: %s"), bUseSocketProjection ? TEXT("YES") : TEXT("NO"));
    UE_LOG(LogTemp, Warning, TEXT("Target Socket Name: %s"), *TargetSocketName.ToString());
    UE_LOG(LogTemp, Warning, TEXT("Socket Offset: (%.1f, %.1f, %.1f)"), SocketOffset.X, SocketOffset.Y, SocketOffset.Z);
    UE_LOG(LogTemp, Warning, TEXT("PlayerController available: %s"), OwnerPlayerController ? TEXT("YES") : TEXT("NO"));
    UE_LOG(LogTemp, Warning, TEXT("========================================"));
}
