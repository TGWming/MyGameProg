    FCameraStateConfig Config = GetExplorationConfig();
    Config.StateBase.Identity. StateName = FName("DarkSoulsStyle");

    // 较慢、更审慎的相机响应
    Config. StateBase.Position. FollowLagSpeed = 6.0f;

FCollisionParams USoulsCameraPresets:: GetDefaultCollisionConfig()
{
    FCollisionParams Config;

    // Detection - 检测配置
    Config. Detection.CameraProbeRadius = 12.0f;
    Config.Detection. MinCameraDistance = 50.0f;
    Config. Detection.CollisionPadding = 10.0f;
    Config.Detection. bUseSphereTrace = true;
    Config.Detection. bUseMultiRay = false;
    Config.Detection.bUsePredictive = true;

    // Response - 响应配置
    Config.Response.bEnablePullIn = true;
    Config.Response.bEnableSlide = false;
    Config.Response.bEnableRotate = false;
    Config. Response.bEnableHybrid = false;
    Config. Response.PullInSpeed = 500.0f;
    Config.Response.bInstantPullIn = false;
    Config.Response.SoftLimitDistance = 100.0f;

    // Occlusion - 遮挡配置
    Config. Occlusion.bEnableCharacterFade = true;
    Config.Occlusion.CharacterFadeStartDistance = 150.0f;
    Config.Occlusion.CharacterFadeEndDistance = 50.0f;
    Config.Occlusion.CharacterMinAlpha = 0.1f;
    Config.Occlusion.CharacterFadeSpeed = 5.0f;
    Config. Occlusion.bEnsureTargetVisible = true;
    Config.Occlusion.bEnableTransparencyProbe = false;
    Config. Occlusion.bEnableSmartReframe = false;

    // Recovery - 恢复配置
    Config.Recovery.RecoveryMode = ECameraRecoveryMode:: Smooth;
    Config. Recovery.RecoverySpeed = 200.0f;
    Config.Recovery. RecoveryDelay = 0.2f;
    Config.Recovery. RecoverySpeedMultiplier = 1.0f;

    // Special - 特殊场景配置
    Config.Special.bEnableTightSpaceHandling = true;
    Config. Special.bEnableWaterHandling = true;
    Config.Special.bEnableDynamicObstacleHandling = true;

    return Config;
}

FCollisionParams USoulsCameraPresets::GetAggressiveCollisionConfig()
