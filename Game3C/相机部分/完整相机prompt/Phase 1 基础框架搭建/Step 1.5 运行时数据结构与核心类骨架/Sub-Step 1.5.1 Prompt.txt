```cpp
// Copyright Your Company.       All Rights Reserved.   

#pragma once

#include "CoreMinimal.h"
#include "SoulsCameraTypes.h"
#include "SoulsCameraRuntimeTypes.generated.h"

/**
 * Runtime data structures for the camera pipeline
 * These structures are used to pass data between pipeline stages
 */

//========================================
// FCameraInputContext - Pipeline Input (Stage 1 Output)
//========================================
/**
 * Input context gathered at the start of each frame
 * Contains all information needed for camera calculations
 * Populated by Stage 1 (InputGather) and consumed by all subsequent stages
 */
USTRUCT(BlueprintType)
struct FCameraInputContext
{
    GENERATED_BODY()

    //--- Time ---
    
    /** Delta time for this frame */
    UPROPERTY(BlueprintReadOnly, Category = "Time")
    float DeltaTime;

    /** Total game time */
    UPROPERTY(BlueprintReadOnly, Category = "Time")
    float GameTime;

    //--- Player Input ---
    
    /** Camera rotation input from player (Yaw, Pitch) */
    UPROPERTY(BlueprintReadOnly, Category = "Input")
    FVector2D CameraInput;

    /** Raw input magnitude (0-1) */
    UPROPERTY(BlueprintReadOnly, Category = "Input")
    float InputMagnitude;

    /** Time since last input */
    UPROPERTY(BlueprintReadOnly, Category = "Input")
    float TimeSinceLastInput;

    //--- Character State ---
    
    /** Character world location */
    UPROPERTY(BlueprintReadOnly, Category = "Character")
    FVector CharacterLocation;

    /** Character world rotation */
    UPROPERTY(BlueprintReadOnly, Category = "Character")
    FRotator CharacterRotation;

    /** Character velocity */
    UPROPERTY(BlueprintReadOnly, Category = "Character")
    FVector CharacterVelocity;

    /** Character movement speed (horizontal) */
    UPROPERTY(BlueprintReadOnly, Category = "Character")
    float CharacterSpeed;

    /** Character is currently moving */
    UPROPERTY(BlueprintReadOnly, Category = "Character")
    bool bIsMoving;

    /** Character is sprinting */
    UPROPERTY(BlueprintReadOnly, Category = "Character")
    bool bIsSprinting;

    /** Character is crouching */
    UPROPERTY(BlueprintReadOnly, Category = "Character")
    bool bIsCrouching;

    /** Character is in air (jumping/falling) */
    UPROPERTY(BlueprintReadOnly, Category = "Character")
    bool bIsInAir;

    /** Character current health ratio (0-1) */
    UPROPERTY(BlueprintReadOnly, Category = "Character")
    float HealthRatio;

    //--- Combat State ---
    
    /** Currently in combat */
    UPROPERTY(BlueprintReadOnly, Category = "Combat")
    bool bIsInCombat;

    /** Has a locked target */
    UPROPERTY(BlueprintReadOnly, Category = "Combat")
    bool bHasTarget;

    /** Currently locked target actor (can be null) */
    UPROPERTY(BlueprintReadOnly, Category = "Combat")
    TWeakObjectPtr<AActor> TargetActor;

    /** Target world location (valid only if bHasTarget) */
    UPROPERTY(BlueprintReadOnly, Category = "Combat")
    FVector TargetLocation;

    /** Distance to target */
    UPROPERTY(BlueprintReadOnly, Category = "Combat")
    float TargetDistance;

    /** Target bounding box size (for distance adjustment) */
    UPROPERTY(BlueprintReadOnly, Category = "Combat")
    float TargetSize;

    /** Target is a boss */
    UPROPERTY(BlueprintReadOnly, Category = "Combat")
    bool bTargetIsBoss;

    //--- Environment ---
    
    /** Camera is in a tight space */
    UPROPERTY(BlueprintReadOnly, Category = "Environment")
    bool bInTightSpace;

    /** Camera is under low ceiling */
    UPROPERTY(BlueprintReadOnly, Category = "Environment")
    bool bUnderLowCeiling;

    /** Near a cliff edge */
    UPROPERTY(BlueprintReadOnly, Category = "Environment")
    bool bNearCliffEdge;

    //--- Previous Frame Data ---
    
    /** Previous frame camera location */
    UPROPERTY(BlueprintReadOnly, Category = "Previous")
    FVector PreviousCameraLocation;

    /** Previous frame camera rotation */
    UPROPERTY(BlueprintReadOnly, Category = "Previous")
    FRotator PreviousCameraRotation;

    /** Constructor with default values */
    FCameraInputContext()
        : DeltaTime(0.0f)
        , GameTime(0.0f)
        , CameraInput(FVector2D::ZeroVector)
        , InputMagnitude(0.0f)
        , TimeSinceLastInput(0.0f)
        , CharacterLocation(FVector::ZeroVector)
        , CharacterRotation(FRotator::ZeroRotator)
        , CharacterVelocity(FVector:: ZeroVector)
        , CharacterSpeed(0.0f)
        , bIsMoving(false)
        , bIsSprinting(false)
        , bIsCrouching(false)
        , bIsInAir(false)
        , HealthRatio(1.0f)
        , bIsInCombat(false)
        , bHasTarget(false)
        , TargetActor(nullptr)
        , TargetLocation(FVector:: ZeroVector)
        , TargetDistance(0.0f)
        , TargetSize(100.0f)
        , bTargetIsBoss(false)
        , bInTightSpace(false)
        , bUnderLowCeiling(false)
        , bNearCliffEdge(false)
        , PreviousCameraLocation(FVector::ZeroVector)
        , PreviousCameraRotation(FRotator::ZeroRotator)
    {}

    /** Reset to default values */
    void Reset()
    {
        *this = FCameraInputContext();
    }

    /** Check if we have valid target data */
    FORCEINLINE bool HasValidTarget() const
    {
        return bHasTarget && TargetActor.IsValid();
    }
};

//========================================

//========================================
```
