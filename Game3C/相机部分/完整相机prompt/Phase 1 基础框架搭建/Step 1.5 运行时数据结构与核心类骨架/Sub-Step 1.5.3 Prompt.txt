//========================================
// FModuleOutput - Single Module Output (Stage 3)

/**
 * Output from a single camera module
 * Multiple module outputs are collected and blended in Stage 5
 */
USTRUCT(BlueprintType)
struct FModuleOutput
{
    GENERATED_BODY()

    //--- Module Identity ---
    
    /** Type of module that produced this output */
    UPROPERTY(BlueprintReadOnly, Category = "Module")
    ECameraModuleType ModuleType;

    /** Module priority (higher = processed later, can override) */
    UPROPERTY(BlueprintReadOnly, Category = "Module")
    int32 Priority;

    /** How this output blends with others */
    UPROPERTY(BlueprintReadOnly, Category = "Module")
    EBlendPolicy BlendPolicy;

    /** Blend weight (0-1) */
    UPROPERTY(BlueprintReadOnly, Category = "Module")
    float Weight;

    //--- Output Values (Optional - only set if module affects this) ---
    
    /** Position offset to apply */
    UPROPERTY(BlueprintReadWrite, Category = "Output")
    FVector PositionOffset;

    /** Has valid position output */
    UPROPERTY(BlueprintReadOnly, Category = "Output")
    bool bHasPositionOutput;

    /** Rotation to apply */
    UPROPERTY(BlueprintReadWrite, Category = "Output")
    FRotator RotationOutput;

    /** Has valid rotation output */
    UPROPERTY(BlueprintReadOnly, Category = "Output")
    bool bHasRotationOutput;

    /** Distance value or offset */
    UPROPERTY(BlueprintReadWrite, Category = "Output")
    float DistanceOutput;

    /** Has valid distance output */
    UPROPERTY(BlueprintReadOnly, Category = "Output")
    bool bHasDistanceOutput;

    /** Distance is multiplier (true) or absolute (false) */
    UPROPERTY(BlueprintReadOnly, Category = "Output")
    bool bDistanceIsMultiplier;

    /** FOV value or offset */
    UPROPERTY(BlueprintReadWrite, Category = "Output")
    float FOVOutput;

    /** Has valid FOV output */
    UPROPERTY(BlueprintReadOnly, Category = "Output")
    bool bHasFOVOutput;

    /** Socket offset to apply */
    UPROPERTY(BlueprintReadWrite, Category = "Output")
    FVector SocketOffsetOutput;

    /** Has valid socket offset output */
    UPROPERTY(BlueprintReadOnly, Category = "Output")
    bool bHasSocketOffsetOutput;

    /** Constructor */
    FModuleOutput()
        : ModuleType(ECameraModuleType::None)
        , Priority(0)
        , BlendPolicy(EBlendPolicy::Blend)
        , Weight(1.0f)
        , PositionOffset(FVector:: ZeroVector)
        , bHasPositionOutput(false)
        , RotationOutput(FRotator::ZeroRotator)
        , bHasRotationOutput(false)
        , DistanceOutput(0.0f)
        , bHasDistanceOutput(false)
        , bDistanceIsMultiplier(false)
        , FOVOutput(0.0f)
        , bHasFOVOutput(false)
        , SocketOffsetOutput(FVector::ZeroVector)
        , bHasSocketOffsetOutput(false)
    {}

    /** Create output with position */
    static FModuleOutput MakePosition(ECameraModuleType Type, const FVector& Position, int32 InPriority = 100, EBlendPolicy Policy = EBlendPolicy::Blend)
    {
        FModuleOutput Output;
        Output.  ModuleType = Type;
        Output. Priority = InPriority;
        Output.  BlendPolicy = Policy;
        Output.  PositionOffset = Position;
        Output. bHasPositionOutput = true;
        return Output;
    }

    /** Create output with rotation */
    static FModuleOutput MakeRotation(ECameraModuleType Type, const FRotator& Rotation, int32 InPriority = 100, EBlendPolicy Policy = EBlendPolicy::Blend)
    {
        FModuleOutput Output;
        Output.   ModuleType = Type;
        Output. Priority = InPriority;
        Output.  BlendPolicy = Policy;
        Output.  RotationOutput = Rotation;
        Output.bHasRotationOutput = true;
        return Output;
    }

    /** Create output with distance */
    static FModuleOutput MakeDistance(ECameraModuleType Type, float Distance, bool bIsMultiplier = false, int32 InPriority = 100, EBlendPolicy Policy = EBlendPolicy::Blend)
    {
        FModuleOutput Output;
        Output.  ModuleType = Type;
        Output. Priority = InPriority;
        Output.  BlendPolicy = Policy;
        Output.  DistanceOutput = Distance;
        Output.bHasDistanceOutput = true;
        Output.bDistanceIsMultiplier = bIsMultiplier;
        return Output;
    }

    /** Create output with FOV */
    static FModuleOutput MakeFOV(ECameraModuleType Type, float FOV, int32 InPriority = 100, EBlendPolicy Policy = EBlendPolicy::Additive)
    {
        FModuleOutput Output;
        Output.ModuleType = Type;
        Output.Priority = InPriority;
        Output. BlendPolicy = Policy;
        Output.   FOVOutput = FOV;
        Output.bHasFOVOutput = true;
        return Output;
    }
};

//========================================
// FModifierOutput - Single Modifier Output (Stage 4)
//========================================
/**
 * Output from a single camera modifier
 * Modifiers are temporary effects that overlay on the base camera
 */
USTRUCT(BlueprintType)
struct FModifierOutput
{
    GENERATED_BODY()

    //--- Modifier Identity ---
    
    /** Type of modifier that produced this output */
    UPROPERTY(BlueprintReadOnly, Category = "Modifier")
    ECameraModifierType ModifierType;

    /** Current modifier state */
    UPROPERTY(BlueprintReadOnly, Category = "Modifier")
    EModifierState State;

    /** Current blend weight (0-1, varies with lifecycle) */
    UPROPERTY(BlueprintReadOnly, Category = "Modifier")
    float CurrentWeight;

    /** Time remaining in this modifier */
    UPROPERTY(BlueprintReadOnly, Category = "Modifier")
    float TimeRemaining;

    //--- Transform Effects ---
    
    /** Position shake/offset */
    UPROPERTY(BlueprintReadWrite, Category = "Effect")
    FVector PositionEffect;

    /** Rotation shake/offset */
    UPROPERTY(BlueprintReadWrite, Category = "Effect")
    FRotator RotationEffect;

    /** Distance offset */
    UPROPERTY(BlueprintReadWrite, Category = "Effect")
    float DistanceEffect;

    /** FOV offset */
    UPROPERTY(BlueprintReadWrite, Category = "Effect")
    float FOVEffect;

    //--- Post Process Effects ---
    
    /** Vignette intensity addition */
    UPROPERTY(BlueprintReadWrite, Category = "PostProcess")
    float VignetteEffect;

    /** Color tint effect */
    UPROPERTY(BlueprintReadWrite, Category = "PostProcess")
    FLinearColor ColorTintEffect;

    /** Saturation multiplier (1 = normal) */
    UPROPERTY(BlueprintReadWrite, Category = "PostProcess")
    float SaturationEffect;

    //--- Special Effects ---
    
    /** Time dilation effect (1 = normal speed) */
    UPROPERTY(BlueprintReadWrite, Category = "Special")
    float TimeDilationEffect;

    /** Should override camera completely (cinematic) */
    UPROPERTY(BlueprintReadOnly, Category = "Special")
    bool bOverrideCamera;

    /** Override camera transform (only if bOverrideCamera) */
    UPROPERTY(BlueprintReadWrite, Category = "Special")
    FTransform OverrideTransform;

    /** Constructor */
    FModifierOutput()
        : ModifierType(ECameraModifierType::None)
        , State(EModifierState:: Inactive)
        , CurrentWeight(0.0f)
        , TimeRemaining(0.0f)
        , PositionEffect(FVector::ZeroVector)
        , RotationEffect(FRotator::ZeroRotator)
        , DistanceEffect(0.0f)
        , FOVEffect(0.0f)
        , VignetteEffect(0.0f)
        , ColorTintEffect(FLinearColor:: White)
        , SaturationEffect(1.0f)
        , TimeDilationEffect(1.0f)
        , bOverrideCamera(false)
        , OverrideTransform(FTransform::Identity)
    {}

    /** Check if modifier is currently active */
    FORCEINLINE bool IsActive() const
    {
        return State != EModifierState::Inactive && CurrentWeight > 0.0f;
    }

    /** Check if this is a camera override modifier */
    FORCEINLINE bool IsOverride() const
    {
        return bOverrideCamera && IsActive();
    }
};
