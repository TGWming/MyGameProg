#include "SoulsCameraStateMachine.h"

USoulsCameraStateMachine::USoulsCameraStateMachine()
    : StatesDataTable(nullptr)
    , CurrentStateName(NAME_None)
    , PreviousStateName(NAME_None)
    , BlendAlpha(1.0f)
    , TransitionTimeRemaining(0.0f)
    , TransitionTotalTime(0.0f)
    , CurrentBlendType(ECameraBlendType::SmoothStep)
    , bIsInitialized(false)
{
}

void USoulsCameraStateMachine::Initialize(UDataTable* InStatesDataTable)
{
    StatesDataTable = InStatesDataTable;
    
    if (StatesDataTable)
    {
        CacheStateConfigs();
        
        // Set default state
        if (CachedConfigs.Num() > 0)
        {
            // Try to find "Explore_Default" or use first state
            if (CachedConfigs. Contains(FName("Explore_Default")))
            {
                CurrentStateName = FName("Explore_Default");
            }
            else
            {
                // Use first available state
                for (const auto& Pair : CachedConfigs)
                {
                    CurrentStateName = Pair.Key;
                    break;
                }
            }
        }
        
        bIsInitialized = true;
        UE_LOG(LogTemp, Log, TEXT("SoulsCameraStateMachine:    Initialized with %d states"), CachedConfigs. Num());
    }
    else
    {
        UE_LOG(LogTemp, Warning, TEXT("SoulsCameraStateMachine:  Initialize called with null DataTable"));
        bIsInitialized = false;
    }
}

bool USoulsCameraStateMachine::IsInitialized() const
{
    return bIsInitialized;
}

bool USoulsCameraStateMachine::RequestStateChange(FName NewStateName, bool bForce)
{
    if (!bIsInitialized)
    {
        return false;
    }

    // Check if state exists
    if (!CachedConfigs. Contains(NewStateName))
    {
        UE_LOG(LogTemp, Warning, TEXT("SoulsCameraStateMachine:  State '%s' does not exist"), *NewStateName. ToString());
        return false;
    }

    // Already in this state
    if (CurrentStateName == NewStateName && !bForce)
    {
        return true;
    }

    // Get new state config for transition settings
    const FCameraStateConfig& NewConfig = CachedConfigs[NewStateName];

    // Store previous state
    PreviousStateName = CurrentStateName;
    CurrentStateName = NewStateName;

    // Setup transition
    TransitionTotalTime = NewConfig.StateBase. Transition.BlendInTime;
    TransitionTimeRemaining = TransitionTotalTime;
    CurrentBlendType = NewConfig.StateBase. Transition.BlendType;
    BlendAlpha = (TransitionTotalTime > 0.0f) ? 0.0f : 1.0f;

    // Broadcast state change
    OnStateChanged.Broadcast(PreviousStateName, CurrentStateName);

    UE_LOG(LogTemp, Log, TEXT("SoulsCameraStateMachine:  State changed from '%s' to '%s'"), 
        *PreviousStateName.ToString(), *CurrentStateName.ToString());

    return true;
}

FName USoulsCameraStateMachine::GetCurrentStateName() const
{
    return CurrentStateName;
}

FName USoulsCameraStateMachine::GetPreviousStateName() const
{
    return PreviousStateName;
}

bool USoulsCameraStateMachine::GetCurrentStateConfig(FCameraStateConfig& OutConfig) const
{
    return GetStateConfig(CurrentStateName, OutConfig);
}

bool USoulsCameraStateMachine::GetStateConfig(FName StateName, FCameraStateConfig& OutConfig) const
{
    if (const FCameraStateConfig* Config = CachedConfigs. Find(StateName))
    {
        OutConfig = *Config;
        return true;
    }
    return false;
}

bool USoulsCameraStateMachine::DoesStateExist(FName StateName) const
{
    return CachedConfigs.Contains(StateName);
}

TArray<FName> USoulsCameraStateMachine::GetStatesInCategory(ECameraStateCategory Category) const
{
    TArray<FName> Result;
    
    for (const auto& Pair : CachedConfigs)
    {
        if (Pair.Value.StateBase.Identity.Category == Category)
        {
            Result.Add(Pair.Key);
        }
    }
    
    return Result;
}

ECameraStateCategory USoulsCameraStateMachine::GetCurrentCategory() const
{
    if (const FCameraStateConfig* Config = CachedConfigs.Find(CurrentStateName))
    {
        return Config->StateBase. Identity.Category;
    }
    return ECameraStateCategory:: None;
}

bool USoulsCameraStateMachine::IsInCategory(ECameraStateCategory Category) const
{
    return GetCurrentCategory() == Category;
}

void USoulsCameraStateMachine::UpdateTransition(float DeltaTime)
{
    if (TransitionTimeRemaining > 0.0f)
    {
        TransitionTimeRemaining -= DeltaTime;
        
        if (TransitionTimeRemaining <= 0.0f)
        {
            TransitionTimeRemaining = 0.0f;
            BlendAlpha = 1.0f;
            OnTransitionComplete.Broadcast(CurrentStateName);
        }
        else if (TransitionTotalTime > 0.0f)
        {
            float LinearAlpha = 1.0f - (TransitionTimeRemaining / TransitionTotalTime);
            BlendAlpha = CalculateBlendAlpha(LinearAlpha, CurrentBlendType);
        }
    }
}

float USoulsCameraStateMachine::GetBlendAlpha() const
{
    return BlendAlpha;
}

bool USoulsCameraStateMachine::IsInTransition() const
{
    return TransitionTimeRemaining > 0.0f;
}

bool USoulsCameraStateMachine::GetBlendedConfig(FCameraStateConfig& OutConfig) const
{
    if (! bIsInitialized)
    {
        return false;
    }

    // If not in transition, just return current config
    if (!IsInTransition() || PreviousStateName == NAME_None)
    {
        return GetCurrentStateConfig(OutConfig);
    }

    // Get both configs
    FCameraStateConfig PreviousConfig, CurrentConfig;
    if (! GetStateConfig(PreviousStateName, PreviousConfig) || 
        !GetStateConfig(CurrentStateName, CurrentConfig))
    {
        return GetCurrentStateConfig(OutConfig);
    }

    // For now, return current config (full blending will be implemented in Stage 5)
    // This is a simplified version - actual parameter blending is complex
    OutConfig = CurrentConfig;
    return true;
}

void USoulsCameraStateMachine::EvaluateState(const FCameraInputContext& Context)
{
    if (!bIsInitialized)
    {
        return;
    }

    // Basic state evaluation logic
    // This can be extended with more sophisticated rules
    
    FName DesiredState = CurrentStateName;

    // Priority-based evaluation
    if (Context.bHasTarget && Context.bTargetIsBoss)
    {
        DesiredState = FName("Boss_Default");
    }
    else if (Context.bHasTarget)
    {
        DesiredState = FName("LockOn_Default");
    }
    else if (Context.bIsInCombat)
    {
        DesiredState = FName("Combat_Default");
    }
    else if (Context.bIsSprinting)
    {
        DesiredState = FName("Sprint_Default");
    }
    else
    {
        DesiredState = FName("Explore_Default");
    }

    // Request state change if different
    if (DesiredState != CurrentStateName && DoesStateExist(DesiredState))
    {
        RequestStateChange(DesiredState);
    }
}

void USoulsCameraStateMachine:: CacheStateConfigs()
{
    CachedConfigs. Empty();

    if (! StatesDataTable)
    {
        return;
    }

    // Get all row names
    TArray<FName> RowNames = StatesDataTable->GetRowNames();

    for (const FName& RowName : RowNames)
    {
        FCameraStateConfig* Config = StatesDataTable->FindRow<FCameraStateConfig>(RowName, TEXT(""));
        if (Config)
        {
            // Use row name as state name if Identity.StateName is not set
            if (Config->StateBase.Identity.StateName == NAME_None)
            {
                Config->StateBase.Identity.StateName = RowName;
            }
            CachedConfigs. Add(RowName, *Config);
        }
    }
}

float USoulsCameraStateMachine::CalculateBlendAlpha(float LinearAlpha, ECameraBlendType BlendType) const
{
    return ApplyBlendCurve(LinearAlpha, BlendType);
}

float USoulsCameraStateMachine:: ApplyBlendCurve(float Alpha, ECameraBlendType BlendType)
{
    Alpha = FMath:: Clamp(Alpha, 0.0f, 1.0f);

    switch (BlendType)
    {
        case ECameraBlendType::Cut: 
            return (Alpha >= 1.0f) ? 1.0f : 0.0f;

        case ECameraBlendType::Linear:
            return Alpha;

        case ECameraBlendType:: SmoothStep: 
            return FMath::SmoothStep(0.0f, 1.0f, Alpha);

        case ECameraBlendType::EaseIn:
            return Alpha * Alpha;

        case ECameraBlendType::EaseOut: 
            return 1.0f - FMath::Pow(1.0f - Alpha, 2.0f);

        case ECameraBlendType::EaseInOut:
            return (Alpha < 0.5f) 
                ?  2.0f * Alpha * Alpha 
                : 1.0f - FMath::Pow(-2.0f * Alpha + 2.0f, 2.0f) / 2.0f;

        case ECameraBlendType:: Cubic:
            return Alpha * Alpha * Alpha;

        case ECameraBlendType::Spring:
            // Simple spring approximation
            return 1.0f - FMath::Cos(Alpha * PI * 0.5f);

        case ECameraBlendType:: Custom:
            // Custom curve would need additional handling
            return Alpha;

        default: 
            return Alpha;
    }
}
