//========================================
// D1: Detection Parameters (6 params)
//========================================
USTRUCT(BlueprintType)
struct FCollisionParams_Detection
{
    GENERATED_BODY()

    /** Main collision probe radius */
    UPROPERTY(EditAnywhere, BlueprintReadWrite, Category = "Detection", meta = (ClampMin = "5.0", ClampMax = "50.0"))
    float ProbeRadius;

    /** Number of rays for multi-ray detection */
    UPROPERTY(EditAnywhere, BlueprintReadWrite, Category = "Detection", meta = (ClampMin = "3", ClampMax = "12"))
    int32 MultiRayCount;

    /** Collision channel to use */
    UPROPERTY(EditAnywhere, BlueprintReadWrite, Category = "Detection")
    TEnumAsByte<ECollisionChannel> CollisionChannel;

    /** Minimum distance from character to start detection */
    UPROPERTY(EditAnywhere, BlueprintReadWrite, Category = "Detection", meta = (ClampMin = "0.0", ClampMax = "100.0"))
    float DetectionStartOffset;

    /** Detection frequency per second */
    UPROPERTY(EditAnywhere, BlueprintReadWrite, Category = "Detection", meta = (ClampMin = "10. 0", ClampMax = "120.0"))
    float DetectionRate;

    /** Use complex collision for detailed geometry */
    UPROPERTY(EditAnywhere, BlueprintReadWrite, Category = "Detection")
    bool bUseComplexCollision;

    FCollisionParams_Detection()
        : ProbeRadius(12.0f)
        , MultiRayCount(5)
        , CollisionChannel(ECC_Camera)
        , DetectionStartOffset(20.0f)
        , DetectionRate(60.0f)
        , bUseComplexCollision(false)
    {}
};

//========================================
// D2: Response Parameters (7 params)
//========================================
USTRUCT(BlueprintType)
struct FCollisionParams_Response
{
    GENERATED_BODY()

    /** Speed to pull camera in on collision */
    UPROPERTY(EditAnywhere, BlueprintReadWrite, Category = "Response", meta = (ClampMin = "1.0", ClampMax = "50.0"))
    float PullInSpeed;

    /** Minimum distance camera can be pulled to */
    UPROPERTY(EditAnywhere, BlueprintReadWrite, Category = "Response", meta = (ClampMin = "20.0", ClampMax = "200.0"))
    float MinPullInDistance;

    /** Enable sliding along surfaces */
    UPROPERTY(EditAnywhere, BlueprintReadWrite, Category = "Response")
    bool bEnableSlide;

    /** Slide friction coefficient */
    UPROPERTY(EditAnywhere, BlueprintReadWrite, Category = "Response", meta = (ClampMin = "0.0", ClampMax = "1.0", EditCondition = "bEnableSlide"))
    float SlideFriction;

    /** FOV increase when camera is pulled in */
    UPROPERTY(EditAnywhere, BlueprintReadWrite, Category = "Response", meta = (ClampMin = "0.0", ClampMax = "30.0"))
    float PullInFOVCompensation;

    /** Enable instant snap in emergency situations */
    UPROPERTY(EditAnywhere, BlueprintReadWrite, Category = "Response")
    bool bEnableInstantSnap;

    /** Distance threshold for instant snap */
    UPROPERTY(EditAnywhere, BlueprintReadWrite, Category = "Response", meta = (ClampMin = "50.0", ClampMax = "200.0", EditCondition = "bEnableInstantSnap"))
    float InstantSnapThreshold;

    FCollisionParams_Response()
        : PullInSpeed(15.0f)
        , MinPullInDistance(50.0f)
        , bEnableSlide(true)
        , SlideFriction(0.3f)
        , PullInFOVCompensation(10.0f)
        , bEnableInstantSnap(true)
        , InstantSnapThreshold(100.0f)
    {}
};

//========================================
// D3: Occlusion Parameters (6 params)
//========================================
USTRUCT(BlueprintType)
struct FCollisionParams_Occlusion
{
    GENERATED_BODY()

    /** Enable occluder fading */
    UPROPERTY(EditAnywhere, BlueprintReadWrite, Category = "Occlusion")
    bool bEnableOccluderFade;

    /** Fade speed for occluding objects */
    UPROPERTY(EditAnywhere, BlueprintReadWrite, Category = "Occlusion", meta = (ClampMin = "1.0", ClampMax = "20.0", EditCondition = "bEnableOccluderFade"))
    float OccluderFadeSpeed;

    /** Minimum opacity for faded occluders */
    UPROPERTY(EditAnywhere, BlueprintReadWrite, Category = "Occlusion", meta = (ClampMin = "0.0", ClampMax = "0.5", EditCondition = "bEnableOccluderFade"))
    float OccluderMinOpacity;

    /** Use dither fade instead of alpha fade */
    UPROPERTY(EditAnywhere, BlueprintReadWrite, Category = "Occlusion")
    bool bUseDitherFade;

    /** Hide occluders completely instead of fading */
    UPROPERTY(EditAnywhere, BlueprintReadWrite, Category = "Occlusion")
    bool bHideOccluders;

    /** Maximum number of occluders to track */
    UPROPERTY(EditAnywhere, BlueprintReadWrite, Category = "Occlusion", meta = (ClampMin = "1", ClampMax = "20"))
    int32 MaxTrackedOccluders;

    FCollisionParams_Occlusion()
        : bEnableOccluderFade(true)
        , OccluderFadeSpeed(8.0f)
        , OccluderMinOpacity(0.2f)
        , bUseDitherFade(false)
        , bHideOccluders(false)
        , MaxTrackedOccluders(5)
    {}
};

//========================================
// D4: Recovery Parameters (5 params)
//========================================
USTRUCT(BlueprintType)
struct FCollisionParams_Recovery
{
    GENERATED_BODY()

    /** Delay before starting recovery after collision ends */
    UPROPERTY(EditAnywhere, BlueprintReadWrite, Category = "Recovery", meta = (ClampMin = "0.0", ClampMax = "3.0"))
    float RecoveryDelay;

    /** Speed to recover to ideal distance */
    UPROPERTY(EditAnywhere, BlueprintReadWrite, Category = "Recovery", meta = (ClampMin = "1.0", ClampMax = "20.0"))
    float RecoverySpeed;

    /** Use stepped recovery (discrete increments) */
    UPROPERTY(EditAnywhere, BlueprintReadWrite, Category = "Recovery")
    bool bUseSteppedRecovery;

    /** Step size for stepped recovery */
    UPROPERTY(EditAnywhere, BlueprintReadWrite, Category = "Recovery", meta = (ClampMin = "10.0", ClampMax = "100.0", EditCondition = "bUseSteppedRecovery"))
    float RecoveryStepSize;

    /** Abort recovery if new collision detected */
    UPROPERTY(EditAnywhere, BlueprintReadWrite, Category = "Recovery")
    bool bAbortRecoveryOnNewCollision;

    FCollisionParams_Recovery()
        : RecoveryDelay(0.5f)
        , RecoverySpeed(5.0f)
        , bUseSteppedRecovery(false)
        , RecoveryStepSize(25.0f)
        , bAbortRecoveryOnNewCollision(true)
    {}
};

//========================================
// D5: Special Situation Parameters (6 params)
//========================================
USTRUCT(BlueprintType)
struct FCollisionParams_Special
{
    GENERATED_BODY()

    /** Width threshold to detect tight space (cm) */
    UPROPERTY(EditAnywhere, BlueprintReadWrite, Category = "Special|TightSpace", meta = (ClampMin = "100.0", ClampMax = "500.0"))
    float TightSpaceThreshold;

    /** Distance reduction in tight spaces */
    UPROPERTY(EditAnywhere, BlueprintReadWrite, Category = "Special|TightSpace", meta = (ClampMin = "0.0", ClampMax = "200.0"))
    float TightSpaceDistanceReduction;

    /** Height threshold for low ceiling detection */
    UPROPERTY(EditAnywhere, BlueprintReadWrite, Category = "Special|LowCeiling", meta = (ClampMin = "150.0", ClampMax = "400.0"))
    float LowCeilingThreshold;

    /** Pitch adjustment in low ceiling areas */
    UPROPERTY(EditAnywhere, BlueprintReadWrite, Category = "Special|LowCeiling", meta = (ClampMin = "0.0", ClampMax = "45.0"))
    float LowCeilingPitchAdjust;

    /** Enable cliff edge detection */
    UPROPERTY(EditAnywhere, BlueprintReadWrite, Category = "Special|Cliff")
    bool bEnableCliffDetection;

    /** Distance to look ahead for cliff edges */
    UPROPERTY(EditAnywhere, BlueprintReadWrite, Category = "Special|Cliff", meta = (ClampMin = "50.0", ClampMax = "300.0", EditCondition = "bEnableCliffDetection"))
    float CliffDetectionDistance;

    FCollisionParams_Special()
        : TightSpaceThreshold(250.0f)
        , TightSpaceDistanceReduction(100.0f)
        , LowCeilingThreshold(250.0f)
        , LowCeilingPitchAdjust(15.0f)
        , bEnableCliffDetection(true)
        , CliffDetectionDistance(150.0f)
    {}
};
