```cpp
/**
 * Blend policy for module/modifier outputs
 * Determines how multiple outputs are combined in the pipeline
 * 混合策略枚举 - 决定多个输出如何合并
 */
UENUM(BlueprintType)
enum class EBlendPolicy : uint8
{
    // Completely replace previous value
    Override = 0                        UMETA(DisplayName = "覆盖 (Override)"),
    
    // Add to previous value (e. g., position offset)
    Additive                            UMETA(DisplayName = "叠加 (Additive)"),
    
    // Multiply with previous value (e.g., distance multiplier)
    Multiplicative                      UMETA(DisplayName = "乘法 (Multiplicative)"),
    
    // Interpolate with previous value using weight
    Blend                               UMETA(DisplayName = "混合 (Blend)"),
    
    // Take minimum of current and previous
    Minimum                             UMETA(DisplayName = "取最小 (Minimum)"),
    
    // Take maximum of current and previous
    Maximum                             UMETA(DisplayName = "取最大 (Maximum)"),
    
    MAX                                 UMETA(Hidden)
};
```

3. EModifierState枚举应包含以下值：

```cpp
/**
 * Modifier lifecycle state
 * Tracks the current phase of a modifier's execution
 * 修改器生命周期状态枚举
 */
UENUM(BlueprintType)
enum class EModifierState : uint8
{
    // Not active
    Inactive = 0                        UMETA(DisplayName = "未激活 (Inactive)"),
    
    // Blending in (weight increasing)
    BlendingIn                          UMETA(DisplayName = "淡入中 (Blending In)"),
    
    // Fully active (weight = 1)
    Active                              UMETA(DisplayName = "激活 (Active)"),
    
    // Blending out (weight decreasing)
    BlendingOut                         UMETA(DisplayName = "淡出中 (Blending Out)"),
    
    // Paused state
    Paused                              UMETA(DisplayName = "暂停 (Paused)"),
    
    MAX                                 UMETA(Hidden)
};
```
