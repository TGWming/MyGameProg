//========================================

UCameraModifier_S05_EnvironmentShake::UCameraModifier_S05_EnvironmentShake()
    : bUseDistanceFalloff(false)
    , MaxShakeDistance(2000.0f)
{
    // Environment shake - powerful and sustained
    PositionAmplitude = 10.0f;
    RotationAmplitude = 2.0f;
    ShakeFrequency = 12.0f; // Low frequency for rumbling feel
}

void UCameraModifier_S05_EnvironmentShake:: OnTriggered(const FModifierTriggerData& TriggerData)
{
    // Generate new random seed for variation
    RandomSeed = FMath:: FRand() * 1000.0f;
    ShakeTime = 0.0f;

    // Check if source location was provided (for distance falloff)
    bUseDistanceFalloff = ! TriggerData. SourceLocation.IsZero();

    // Scale parameters based on intensity
    float IntensityScale = FMath::Clamp(TriggerData. Intensity, 0.3f, 3.0f);
    PositionAmplitude = 10.0f * IntensityScale;
    RotationAmplitude = 2.0f * IntensityScale;

    // Environmental events can affect FOV (explosion shockwave feel)
    FOVEffect = 5.0f * IntensityScale; // Slight FOV increase for impact

    // Configure based on custom tag
    if (TriggerData.CustomTag == FName("Earthquake"))
    {
        ShakeFrequency = 8.0f; // Very low frequency
        PositionAmplitude *= 1.5f;
        MaxShakeDistance = 5000.0f; // Earthquakes affect large area
    }
    else if (TriggerData.CustomTag == FName("Explosion"))
    {
        ShakeFrequency = 15.0f;
        MaxShakeDistance = 1500.0f;
        FOVEffect = 8.0f * IntensityScale; // Stronger FOV punch
    }
    else if (TriggerData.CustomTag == FName("BossAbility"))
    {
        ShakeFrequency = 10.0f;
        MaxShakeDistance = 3000.0f;
    }
    else
    {
        ShakeFrequency = 12.0f;
        MaxShakeDistance = 2000.0f;
    }

    UE_LOG(LogTemp, Verbose, TEXT("S05 EnvironmentShake: Triggered with intensity %.2f, tag: %s"),
        TriggerData. Intensity, *TriggerData. CustomTag.ToString());
}

void UCameraModifier_S05_EnvironmentShake::ComputeEffect(float DeltaTime, const FStageExecutionContext& Context)
{
    // Accumulate shake time
    ShakeTime += DeltaTime;

    // Calculate distance falloff if applicable
    float DistanceMultiplier = 1.0f;
    if (bUseDistanceFalloff)
    {
        DistanceMultiplier = CalculateDistanceFalloff(Context);
    }

    // Get effective amplitude
    float EffectiveAmp = GetEffectiveAmplitude() * DistanceMultiplier;

    // Generate position shake
    PositionEffect = GeneratePositionShake(ShakeTime, ShakeFrequency, PositionAmplitude * EffectiveAmp);

    // Generate rotation shake
    RotationEffect = GenerateRotationShake(ShakeTime, ShakeFrequency, RotationAmplitude * EffectiveAmp);

    // Decay FOV effect over time with distance
    float NormalizedTime = StateTime / ActiveDuration;
    FOVEffect = FOVEffect * (1.0f - NormalizedTime) * DistanceMultiplier;
}

float UCameraModifier_S05_EnvironmentShake::CalculateDistanceFalloff(const FStageExecutionContext& Context) const
{
    if (!bUseDistanceFalloff || MaxShakeDistance <= 0.0f)
    {
        return 1.0f;
    }

    FVector PlayerLocation = Context.InputContext.CharacterLocation;
    FVector SourceLocation = TriggerData.SourceLocation;

    float Distance = FVector:: Dist(PlayerLocation, SourceLocation);

    // Linear falloff with smooth start
    if (Distance >= MaxShakeDistance)
    {
        return 0.0f;
    }

    float NormalizedDistance = Distance / MaxShakeDistance;
    
    // Use smooth falloff curve (stronger near source, gradual falloff)
    float Falloff = 1.0f - FMath::Pow(NormalizedDistance, 0.5f);
    
    return FMath::Clamp(Falloff, 0.0f, 1.0f);
}
