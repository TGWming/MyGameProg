```cpp
void USoulsCameraManager:: CreateSubSystems()
{
	UE_LOG(LogTemp, Log, TEXT("SoulsCameraManager:   Creating sub-systems... "));

	// Create State Machine
	StateMachine = NewObject<USoulsCameraStateMachine>(this);
	if (StateMachine)
	{
		StateMachine->Initialize(this);
	}

	// Create Module Registry
	ModuleRegistry = NewObject<UCameraModuleRegistry>(this);
	if (ModuleRegistry)
	{
		ModuleRegistry->Initialize(this);
	}

	// Create Modifier Manager (26 modifiers)
	ModifierManager = NewObject<UCameraModifierManager>(this);
	if (ModifierManager)
	{
		ModifierManager->Initialize(this);
		UE_LOG(LogTemp, Log, TEXT("SoulsCameraManager:   ModifierManager created with %d modifiers"),
			ModifierManager->GetAllModifiers().Num());
	}

	// Create Pipeline
	Pipeline = NewObject<USoulsCameraPipeline>(this);
	if (Pipeline)
	{
		Pipeline->Initialize(this);
	}

	// Create and register all stages
	CreateAndRegisterStages();

	// CollisionResolver will be created in Phase 5

	UE_LOG(LogTemp, Log, TEXT("SoulsCameraManager:  Sub-systems created successfully"));
	UE_LOG(LogTemp, Log, TEXT("  - StateMachine: %s"), StateMachine ?   TEXT("OK") : TEXT("FAILED"));
	UE_LOG(LogTemp, Log, TEXT("  - ModuleRegistry: %s"), ModuleRegistry ?  TEXT("OK") : TEXT("FAILED"));
	UE_LOG(LogTemp, Log, TEXT("  - ModifierManager:  %s (%d modifiers)"), 
		ModifierManager ? TEXT("OK") : TEXT("FAILED"),
		ModifierManager ? ModifierManager->GetAllModifiers().Num() : 0);
	UE_LOG(LogTemp, Log, TEXT("  - Pipeline:  %s"), Pipeline ?  TEXT("OK") : TEXT("FAILED"));
```

```cpp
void USoulsCameraManager::CreateAndRegisterStages()
{
	if (!Pipeline)
	{
		UE_LOG(LogTemp, Error, TEXT("SoulsCameraManager:   Cannot create stages - Pipeline is null"));
		return;
	}

	UE_LOG(LogTemp, Log, TEXT("SoulsCameraManager:  Creating and registering pipeline stages..."));

	// Stage 1:   Input Collection
	UCameraStage_InputCollection* Stage1 = NewObject<UCameraStage_InputCollection>(this);
	Stage1->Initialize(this);
	Pipeline->RegisterStage(1, Stage1);

	// Stage 2:  State Evaluation
	UCameraStage_StateEvaluation* Stage2 = NewObject<UCameraStage_StateEvaluation>(this);
	Stage2->Initialize(this);
	Pipeline->RegisterStage(2, Stage2);

	// Stage 3: Module Execute
	UCameraStage_ModuleExecute* Stage3 = NewObject<UCameraStage_ModuleExecute>(this);
	Stage3->Initialize(this);
	Pipeline->RegisterStage(3, Stage3);

	// Stage 4: Modifier Apply - Connect to ModifierManager
	Stage4_ModifierApply = NewObject<UCameraStage_ModifierApply>(this);
	Stage4_ModifierApply->Initialize(this);
	Stage4_ModifierApply->SetModifierManager(ModifierManager);  // Connect ModifierManager
	Pipeline->RegisterStage(4, Stage4_ModifierApply);

	// Stage 5: Collision Resolve (placeholder - Phase 5)
	UCameraStage_CollisionResolve* Stage5 = NewObject<UCameraStage_CollisionResolve>(this);
	Stage5->Initialize(this);
	Pipeline->RegisterStage(5, Stage5);

	// Stage 6: Final Blend
	UCameraStage_FinalBlend* Stage6 = NewObject<UCameraStage_FinalBlend>(this);
	Stage6->Initialize(this);
	Pipeline->RegisterStage(6, Stage6);

	// Stage 7: Output Apply
	UCameraStage_OutputApply* Stage7 = NewObject<UCameraStage_OutputApply>(this);
	Stage7->Initialize(this);
	Pipeline->RegisterStage(7, Stage7);

	UE_LOG(LogTemp, Log, TEXT("SoulsCameraManager:   All 7 stages registered"));
}
```

**添加公共接口函数实现：**

```cpp
//========================================
// Modifier System Interface Implementation
//========================================

void USoulsCameraManager:: TriggerCameraShake(ECameraModifierID ShakeID, float Intensity)
{
	if (ModifierManager)
	{
		ModifierManager->TriggerModifierSimple(ShakeID, Intensity);
		UE_LOG(LogTemp, Verbose, TEXT("SoulsCameraManager:   Triggered shake %s with intensity %. 2f"),
			*UEnum::GetValueAsString(ShakeID), Intensity);
	}
}

void USoulsCameraManager::TriggerHitReaction(bool bHeavyHit, float Intensity, FVector HitDirection)
{
	if (ModifierManager)
	{
		ModifierManager->TriggerHitReaction(bHeavyHit, Intensity, HitDirection);
		UE_LOG(LogTemp, Verbose, TEXT("SoulsCameraManager:  Triggered %s hit reaction, intensity %.2f"),
			bHeavyHit ? TEXT("heavy") :   TEXT("light"), Intensity);
	}
}

void USoulsCameraManager::TriggerSlowMotion(float Duration, float TimeDilation)
{
	if (ModifierManager)
	{
		ModifierManager->TriggerSlowMotion(Duration, TimeDilation);
		UE_LOG(LogTemp, Verbose, TEXT("SoulsCameraManager:  Triggered slow motion, duration=%.2f, dilation=%.2f"),
			Duration, TimeDilation);
	}
}

void USoulsCameraManager::TriggerHitStop(float Intensity)
{
	if (ModifierManager)
	{
		ModifierManager->TriggerHitStop(Intensity);
		UE_LOG(LogTemp, Verbose, TEXT("SoulsCameraManager:  Triggered hit stop, intensity=%.2f"), Intensity);
	}
}

void USoulsCameraManager::TriggerLowHealthEffect(float HealthPercent)
{
	if (ModifierManager)
	{
		ModifierManager->TriggerLowHealthEffect(HealthPercent);
		UE_LOG(LogTemp, Verbose, TEXT("SoulsCameraManager:  Triggered low health effect, health=%.0f%%"),
			HealthPercent * 100.0f);
	}
}

void USoulsCameraManager::StopLowHealthEffect()
{
	if (ModifierManager)
	{
		ModifierManager->StopModifier(ECameraModifierID:: Modifier_E01_Effect_LowHealth, false);
		UE_LOG(LogTemp, Verbose, TEXT("SoulsCameraManager:  Stopped low health effect"));
	}
}

void USoulsCameraManager::TriggerDeathCamera(FVector DeathLocation)
{
	if (ModifierManager)
	{
		FModifierTriggerData TriggerData;
		TriggerData. Intensity = 1.0f;
		TriggerData. SourceLocation = DeathLocation;
		
		ModifierManager->TriggerModifier(ECameraModifierID::Modifier_X03_Special_DeathCam, TriggerData);
		UE_LOG(LogTemp, Log, TEXT("SoulsCameraManager:  Triggered death camera at (%.0f, %.0f, %.0f)"),
			DeathLocation.X, DeathLocation.Y, DeathLocation.Z);
	}
}
```

【代码设计说明】

### 1. ModifierManager连接流程

```
SoulsCameraManager初始化：
├── CreateSubSystems()
│   ├── 创建 StateMachine
│   ├── 创建 ModuleRegistry
│   ├── 创建 ModifierManager ← 新增
│   ├── 创建 Pipeline
│   └── 调用 CreateAndRegisterStages()
│
└── CreateAndRegisterStages()
    └── Stage 4 创建时：
        └── Stage4->SetModifierManager(ModifierManager) ← 连接
```

### 2. 公共接口设计

```
SoulsCameraManager 提供的便捷接口：
├── GetModifierManager()       - 获取Manager直接操作
├── TriggerCameraShake()       - 触发抖动
├── TriggerHitReaction()       - 触发受击反应
├── TriggerSlowMotion()        - 触发慢动作
├── TriggerHitStop()           - 触发顿帧
├── TriggerLowHealthEffect()   - 触发低血量效果
├── StopLowHealthEffect()      - 停止低血量效果
└── TriggerDeathCamera()       - 触发死亡相机
```

### 3. 使用示例

```cpp
// 在游戏代码中使用：

// 获取相机管理器
USoulsCameraManager* CameraManager = GetSoulsCameraManager();

// 方式1：使用便捷接口
CameraManager->TriggerHitReaction(true, 1.5f);  // 重击
CameraManager->TriggerSlowMotion(0.5f, 0.3f);   // 0.5秒慢动作

// 方式2：直接使用ModifierManager
UCameraModifierManager* ModMgr = CameraManager->GetModifierManager();
ModMgr->TriggerModifierSimple(ECameraModifierID::Modifier_Z04_Zoom_CriticalHit, 1.0f);
```

### 4. 与蓝图的集成

```
所有公共接口都有UFUNCTION宏：
├── BlueprintCallable - 可在蓝图中调用
├── BlueprintPure - 可在蓝图中作为纯函数
└── Category分类 - 便于在蓝图中查找
```
