```cpp
 * Stage 4: Modifier Application
 * 
 * This stage applies all active camera modifiers to the camera output.
 * Modifiers include shakes, reactions, zooms, effects, cinematics, and specials.
 * 
 * The stage works with ModifierManager which maintains 26 registered modifiers.
 * Each frame, active modifiers are updated and their effects are collected

class MYPROJECT_API UCameraStage_ModifierApply : public UCameraStageBase

	GENERATED_BODY()

	UCameraStage_ModifierApply();

	//========================================
	// Stage Interface
	//========================================

	virtual void Initialize(USoulsCameraManager* InCameraManager) override;
	virtual void Execute(FStageExecutionContext& Context) override;
	virtual FString GetStageName() const override { return TEXT("ModifierApply"); }

	//========================================
	// Modifier Manager Connection
	//========================================

	/**
	 * Set the modifier manager for this stage
	 * Called by SoulsCameraManager during initialization
	 * @param InModifierManager The modifier manager to use
	 */
	UFUNCTION(BlueprintCallable, Category = "Camera|Stage")
	void SetModifierManager(UCameraModifierManager* InModifierManager);

	/**
	 * Get the connected modifier manager
	 * @return The modifier manager, or nullptr if not set
	 */
	UFUNCTION(BlueprintPure, Category = "Camera|Stage")
	UCameraModifierManager* GetModifierManager() const { return ModifierManager; }

	//========================================
	// Modifier Query
	//========================================

	 * Check if any modifiers are currently active
	 * @return True if at least one modifier is active
	 */
	UFUNCTION(BlueprintPure, Category = "Camera|Stage")

	/**
	 * Get the number of currently active modifiers
	 * @return Number of active modifiers
	 */

	int32 GetActiveModifierCount() const;

protected:
	//========================================
	// Internal Methods
	//========================================

	/**
	 * Update all modifiers for this frame
	 * @param Context Current execution context
	 */
	void UpdateModifiers(FStageExecutionContext& Context);

	 * Collect outputs from all active modifiers

	 */
	void CollectModifierOutputs(FStageExecutionContext& Context);
```

## 文件2：CameraStage_ModifierApply.cpp 更新

**完整替换实现：**

```cpp
// Copyright Epic Games, Inc.  All Rights Reserved. 

#include "Camera/Pipeline/CameraStage_ModifierApply.  h"
#include "Camera/Core/SoulsCameraManager. h"
#include "Camera/Modifiers/CameraModifierManager.h"
#include "Camera/Modifiers/CameraModifierBase.h"

UCameraStage_ModifierApply:: UCameraStage_ModifierApply()
	: ModifierManager(nullptr)
{
	StageIndex = 4;
}

void UCameraStage_ModifierApply::Initialize(USoulsCameraManager* InCameraManager)
{
	Super::Initialize(InCameraManager);
	
	UE_LOG(LogTemp, Log, TEXT("Stage4_ModifierApply:   Initialized"));
}

void UCameraStage_ModifierApply:: SetModifierManager(UCameraModifierManager* InModifierManager)
{
	ModifierManager = InModifierManager;
	
	if (ModifierManager)
	{
		UE_LOG(LogTemp, Log, TEXT("Stage4_ModifierApply:  ModifierManager connected with %d modifiers"),
			ModifierManager->GetAllModifiers().Num());
	}
	else
	{
		UE_LOG(LogTemp, Warning, TEXT("Stage4_ModifierApply:   ModifierManager set to nullptr"));
	}
}

void UCameraStage_ModifierApply:: Execute(FStageExecutionContext& Context)
{
	if (!bIsEnabled)
	{
		return;
	}
	
	// Clear previous frame's modifier outputs
	Context.ModifierOutputs.Reset();
	
	// Check if we have a valid modifier manager
	if (!ModifierManager)
	{
		UE_LOG(LogTemp, Warning, TEXT("Stage4_ModifierApply:   No ModifierManager available"));
		return;
	}
	
	// Update all modifiers for this frame
	UpdateModifiers(Context);
	
	// Check for override modifiers first (cinematics, death cam)
	if (HandleOverrideModifiers(Context))
	{
		// Override modifier is active, it takes full control
		return;
	}
	
	// Collect outputs from all active non-override modifiers
	CollectModifierOutputs(Context);
}

bool UCameraStage_ModifierApply:: HasActiveModifiers() const
{
	if (ModifierManager)
	{
		return ModifierManager->HasActiveModifiers();
	}
	return false;
}

int32 UCameraStage_ModifierApply::GetActiveModifierCount() const
{
	if (ModifierManager)
	{
		return ModifierManager->GetActiveModifierCount();
	}
	return 0;
}

void UCameraStage_ModifierApply::UpdateModifiers(FStageExecutionContext& Context)
{
	if (!ModifierManager)
	{
		return;
	}
	
	// Update all modifiers (this updates their internal state)
	ModifierManager->UpdateModifiers(Context. DeltaTime, Context);
	
	// Refresh cached active modifiers list
	CachedActiveModifiers.Reset();
	ModifierManager->GetActiveModifiers(CachedActiveModifiers);
}

void UCameraStage_ModifierApply::CollectModifierOutputs(FStageExecutionContext& Context)
{
	if (!ModifierManager)
	{
		return;
	}
	
	// Get outputs from modifier manager (already sorted by priority)
	ModifierManager->GetActiveModifierOutputs(Context, Context.ModifierOutputs);
	
	// Log active modifiers (verbose)
	if (Context.ModifierOutputs.Num() > 0)
	{
		UE_LOG(LogTemp, Verbose, TEXT("Stage4_ModifierApply:   Collected %d modifier outputs"),
			Context.ModifierOutputs.Num());
		
		for (const FModifierOutput& Output : Context.ModifierOutputs)
		{
			UE_LOG(LogTemp, Verbose, TEXT("  - %s:  Weight=%.2f, FOV=%.1f, Pos=(%.1f,%.1f,%.1f)"),
				*UEnum::GetValueAsString(Output.  ModifierType),
				Output. Weight,
				Output.  FOVOffset,
				Output. PositionOffset.X, Output.PositionOffset.Y, Output.PositionOffset.Z);
		}
	}
}

bool UCameraStage_ModifierApply::HandleOverrideModifiers(FStageExecutionContext& Context)
{
	// Check for override modifiers in the active list
	for (UCameraModifierBase* Modifier : CachedActiveModifiers)
	{
		if (Modifier && Modifier->IsOverrideModifier() && Modifier->IsOverriding())
		{
			// Get the override output
			FModifierOutput OverrideOutput;
			if (Modifier->GetOutput(Context, OverrideOutput))
			{
				// Mark as override in context
				Context. bHasOverrideModifier = true;
				Context. OverrideTransform = Modifier->GetOverrideTransform();
				
				// Still add the output for visual effects (vignette, color, etc.)
				Context.ModifierOutputs.Add(OverrideOutput);
				
				UE_LOG(LogTemp, Verbose, TEXT("Stage4_ModifierApply:  Override modifier active - %s"),
					*Modifier->GetModifierDescription());
				
				return true;
			}
		}
	}
	
	return false;
}
```

【代码设计说明】

### 1. 执行流程

```
Execute()
├── 清空上一帧的ModifierOutputs
├── 检查ModifierManager有效性
├── UpdateModifiers() - 更新所有修改器状态
├── HandleOverrideModifiers() - 检查Override修改器
│   └── 如果有Override，设置Context标记并返回
└── CollectModifierOutputs() - 收集普通修改器输出
```

### 2. Override修改器处理

```
Override修改器（Cinematic/DeathCam）：
├── IsOverrideModifier() = true
├── IsOverriding() = true（激活时）
├── 完全接管相机Transform
└── 其他修改器的位置/旋转效果被忽略
    （但视觉效果如晕影仍然应用）
```

### 3. 与Blend Stage（Stage 5）的协作

```
Stage 4 输出：
├── Context.ModifierOutputs[]  - 所有活跃修改器的输出
├── Context.bHasOverrideModifier - Override标记
└── Context.OverrideTransform - Override变换

Stage 5 处理：
├── 如果 bHasOverrideModifier：
│   └── 直接使用 OverrideTransform
└── 否则：
    └── 混合所有 ModifierOutputs
```
