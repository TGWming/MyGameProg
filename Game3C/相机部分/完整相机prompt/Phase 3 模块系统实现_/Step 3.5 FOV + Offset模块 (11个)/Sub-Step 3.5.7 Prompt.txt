/**
 * ShouldActivate - Determine if impact FOV effect should be active
 * 
 * F06 activates when: 
 * - Module is enabled
 * - Impact effect is currently active (bImpactActive)
 * - Impact intensity is above threshold
 * 
 * Unlike other FOV modules, F06 is triggered externally via TriggerImpact()
 * rather than being based on game state.   This allows precise control over
 * when the impact effect occurs.
 * 
 * Priority 125 (highest among FOV modules), uses Additive policy. 
 * The high priority ensures the impact effect is visible on top of
 * all other FOV adjustments.
 * 
 * Use Cases:
 * - Player takes damage
 * - Player lands a heavy hit
 * - Parry/counter timing
 * - Explosion nearby
 * - Boss attack impacts
 */
bool UCameraModule_F06_Impact_FOV::ShouldActivate(
	const FStageExecutionContext& Context, 
	const FCameraStateConfig& StateConfig) const
{
	// Check if module is enabled
	if (!bIsEnabled)
	{
		return false;
	}
	
	// Check if impact effect is currently active
	if (! bImpactActive)
	{
		return false;
	}
	
	// Check if intensity is above minimum threshold
	// This prevents very small lingering effects
	const float MinIntensityThreshold = 0.01f;
	if (CurrentImpactIntensity <= MinIntensityThreshold)
	{
		return false;
	}
	
	// Impact is active and significant - activate module
	return true;
}

**C.  Compute() 实现：**

```cpp
/**
 * Compute - Calculate impact FOV pulse effect
 * 
 * This module creates a momentary FOV change (typically reduction)
 * when an impact is triggered.   The effect quickly decays back to normal.
 * 
 * Steps:
 * 1. Validate impact is active
 * 2. Update impact timer
 * 3. Check if effect has expired
 * 4. Get impact settings from config
 * 5. Calculate decayed intensity
 * 6. Apply FOV offset
 * 7. Output with Additive policy
 * 
 * Impact Behavior:
 * - Trigger:  FOV instantly reduces (punch-in)
 * - Decay: FOV smoothly returns to normal
 * - Duration:  Typically 0.1-0.3 seconds
 * - Curve: Exponential decay for snappy feel
 * 
 * FOV Change:
 * - Negative offset: Punch-in effect (zoom in feeling)
 * - Positive offset:  Punch-out effect (explosion feeling)
 * - Default: Negative (punch-in)
 * 
 * @param Context The current pipeline execution context
 * @param StateConfig The current camera state configuration
 * @param OutOutput Output structure to fill with computed values
 * @return True if computation succeeded
 */
bool UCameraModule_F06_Impact_FOV::Compute(
	const FStageExecutionContext& Context, 
	const FCameraStateConfig& StateConfig, 
	FModuleOutput& OutOutput)
{
	// Create base output with module identity
	OutOutput = CreateBaseOutput();
	
	// Step 1: Validate impact is active
	if (!bImpactActive)
	{
		OutOutput.  FOVOutput = 0. 0f;
		OutOutput. bHasFOVOutput = true;
		return false;
	}
	
	// Step 2: Update impact timer
	ImpactTimeRemaining -= Context.DeltaTime;
	
	// Step 3: Check if effect has expired
	if (ImpactTimeRemaining <= 0.0f)
	{
		// Effect has ended - reset state
		bImpactActive = false;
		CurrentImpactIntensity = 0.0f;
		ImpactTimeRemaining = 0.0f;
		
		OutOutput. FOVOutput = 0.0f;
		OutOutput. bHasFOVOutput = true;
		return false;
	}
	
	// Step 4: Get impact settings from config
	float ImpactFOVAmount = StateConfig.Module. FOV.ImpactFOVAmount;
	float ImpactDuration = StateConfig.Module.FOV.ImpactFOVDuration;
	
	// Default values if not configured
	if (ImpactFOVAmount <= 0.0f)
	{
		ImpactFOVAmount = 10.0f;  // Default:  10 degrees
	}
	
	if (ImpactDuration <= 0.0f)
	{
		ImpactDuration = 0.15f;  // Default: 0.15 seconds
	}
	
	// Step 5: Calculate decayed intensity using exponential curve
	// This creates a snappy "punch" feel that quickly fades
	float NormalizedTime = ImpactTimeRemaining / ImpactDuration;
	
	// Exponential decay: intensity² for quick falloff
	// At t=1. 0 (start): intensity = 1.0
	// At t=0.5 (half): intensity = 0.25
	// At t=0.0 (end): intensity = 0.0
	float DecayedIntensity = CurrentImpactIntensity * NormalizedTime * NormalizedTime;
	
	// Step 6: Calculate FOV offset
	// Negative for punch-in effect (most common for hits)
	// CurrentImpactIntensity allows scaling the effect
	float FOVOffset = -ImpactFOVAmount * DecayedIntensity;
	
	// Clamp to prevent extreme values
	FOVOffset = FMath:: Clamp(FOVOffset, -25.0f, 10.0f);
	
	// Step 7: Fill output structure
	OutOutput. FOVOutput = FOVOffset;
	OutOutput.bHasFOVOutput = true;
	OutOutput.  Weight = 1.0f;
	
	return true;
}
```

**D. TriggerImpact() 实现：**

```cpp
/**
 * TriggerImpact - Externally trigger an FOV impact pulse
 * 
 * Call this function when the player takes damage, lands a hit,
 * or any other moment that should have FOV feedback.
 * 
 * The intensity parameter scales the effect: 
 * - 0.5:  Light hit (small enemy attack)
 * - 1.0: Normal hit (standard attack)
 * - 1.5: Heavy hit (boss attack, critical hit)
 * - 2.0: Maximum impact (massive damage, near-death)
 * 
 * @param Intensity Strength of the impact effect (0.0 to 2.0)
 */
void UCameraModule_F06_Impact_FOV::TriggerImpact(float Intensity)
{
	// Clamp intensity to valid range
	// Allow values up to 2.0 for extra heavy impacts
	CurrentImpactIntensity = FMath::Clamp(Intensity, 0.0f, 2.0f);
	
	// Activate the impact effect
	bImpactActive = true;
	
	// Set default duration
	// This can be overridden by config in Compute()
	ImpactTimeRemaining = 0.15f;  // 150ms default
	
	// Optional: Log for debugging
	// UE_LOG(LogTemp, Verbose, TEXT("F06 Impact FOV:  Triggered with intensity %. 2f"), Intensity);
}

/**
 * Compute - Calculate impact FOV pulse with exponential decay
 */
bool UCameraModule_F06_Impact_FOV:: Compute(
	const FStageExecutionContext& Context, 
	const FCameraStateConfig& StateConfig, 
	FModuleOutput& OutOutput)
{
	// Create base output
	OutOutput = CreateBaseOutput();
	
	// Validate impact is active
	if (!bImpactActive)
	{
		OutOutput.FOVOutput = 0.0f;
		OutOutput.bHasFOVOutput = true;
		return false;
	}
	
	// Update timer
	ImpactTimeRemaining -= Context.DeltaTime;
	
	// Check if expired
	if (ImpactTimeRemaining <= 0.0f)
	{
		bImpactActive = false;
		CurrentImpactIntensity = 0.0f;
		ImpactTimeRemaining = 0.0f;
		
		OutOutput. FOVOutput = 0.0f;
		OutOutput. bHasFOVOutput = true;
		return false;
	}
	
	// Get config values
	float ImpactFOVAmount = StateConfig. Module.FOV. ImpactFOVAmount;
	float ImpactDuration = StateConfig. Module.FOV. ImpactFOVDuration;
	
	// Default values
	if (ImpactFOVAmount <= 0.0f)
	{
		ImpactFOVAmount = 10.0f;
	}
	
	if (ImpactDuration <= 0.0f)
	{
		ImpactDuration = 0.15f;
	}
	
	// Calculate decayed intensity (exponential)
	float NormalizedTime = ImpactTimeRemaining / ImpactDuration;
	float DecayedIntensity = CurrentImpactIntensity * NormalizedTime * NormalizedTime;
	
	// Calculate FOV offset (negative for punch-in)
	float FOVOffset = -ImpactFOVAmount * DecayedIntensity;
	FOVOffset = FMath::Clamp(FOVOffset, -25.0f, 10.0f);
	
	// Fill output
	OutOutput.FOVOutput = FOVOffset;
	OutOutput.  bHasFOVOutput = true;
	OutOutput.Weight = 1.0f;
	
	return true;
}

/**
 * TriggerImpact - Trigger the FOV impact pulse
 */
void UCameraModule_F06_Impact_FOV::TriggerImpact(float Intensity)
{
	CurrentImpactIntensity = FMath::Clamp(Intensity, 0.0f, 2.0f);
	bImpactActive = true;
	ImpactTimeRemaining = 0.15f;
}

### 1. 外部触发模式

**与其他模块的区别：**

```

### 2. 指数衰减曲线

**为什么使用 t² 而不是线性？**

```
线性衰减：
t=1.0 → intensity=1.0
t=0.5 → intensity=0.5
t=0.0 → intensity=0.0

效果：平均减弱，感觉拖沓

指数衰减（t²）：
t=1.0 → intensity=1.0
t=0.5 → intensity=0.25
t=0.0 → intensity=0.0

效果：快速减弱，感觉干脆

视觉时间线：
0ms: FOV -10°（满强度）
37ms: FOV -2. 5°（已衰减75%）
75ms: FOV -0.6°（几乎消失）
150ms: FOV 0°（完全恢复）

好处：
- "Punch"感觉更强烈
- 快速恢复正常视野
- 不干扰游戏操作
```

### 3. 强度参数的使用

**Intensity参数如何影响效果：**

```cpp
TriggerImpact(0.5f);   // 轻击
// FOVOffset = -10 × 0.5 × t² = 最大-5°

TriggerImpact(1.0f);   // 标准打击
// FOVOffset = -10 × 1.0 × t² = 最大-10°

TriggerImpact(1.5f);   // 重击
// FOVOffset = -10 × 1.5 × t² = 最大-15°

TriggerImpact(2.0f);   // 致命打击
// FOVOffset = -10 × 2.0 × t² = 最大-20°

使用场景：
- 小怪攻击:  Intensity = 0.3-0.5
- 普通攻击:  Intensity = 0.8-1.0
- 重攻击: Intensity = 1.2-1.5
- Boss技能:  Intensity = 1.5-2.0
- 濒死时: Intensity = 2.0
```

### 4. 状态管理

**三个成员变量的作用：**

```cpp
float CurrentImpactIntensity;  // 当前强度（0-2）
bool bImpactActive;             // 效果是否激活
float ImpactTimeRemaining;      // 剩余时间

状态转换：
1. 初始状态：
   Intensity=0, Active=false, Time=0

2. 触发后：
   Intensity=1.0, Active=true, Time=0.15

3. 每帧更新：
   Time -= DeltaTime
   计算衰减后的FOV偏移

4. 结束时：
   Time <= 0
   重置所有状态
```
