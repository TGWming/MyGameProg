```cpp
void UCameraStage_CollisionResolve::Execute(FStageExecutionContext& Context)
{
    if (!bEnabled)
    {
        return;
    }

    // Use CollisionResolver if available
    if (CollisionResolver && CollisionResolver->IsInitialized())
    {
        // Let the resolver handle all collision logic
        CollisionResolver->ResolveCollisions(Context);
        
        // Update resolver with delta time
        CollisionResolver->Update(Context.DeltaTime, Context);
    }
    else
    {
        // Fallback to basic collision check
        PerformBasicCollisionCheck(Context);
    }
}
```

**实现PerformBasicCollisionCheck fallback：**

```cpp
void UCameraStage_CollisionResolve::  PerformBasicCollisionCheck(FStageExecutionContext& Context)
{
    // Basic line trace fallback when CollisionResolver is not available
    UWorld* World = GEngine ?  GEngine->GetWorldFromContextObject(this, EGetWorldErrorMode:: ReturnNull) :   nullptr;
    if (! World)
    {
        return;
    }

    FVector FocusPoint = Context.Output. FocusPoint;
    FVector CameraDirection = -Context.Output.  Rotation.Vector();
    float TargetDistance = Context. Output.  Distance;
    FVector TargetCameraPos = FocusPoint + CameraDirection * TargetDistance;

    FCollisionQueryParams Params;
    Params.  bTraceComplex = false;
    Params. AddIgnoredActor(GetOuter());  // Ignore owner

    FHitResult Hit;
    if (World->LineTraceSingleByChannel(Hit, FocusPoint, TargetCameraPos, ECC_Camera, Params))
    {
        // Collision detected - adjust distance
        float SafeDistance = FMath::Max(Hit.Distance - 10. 0f, 50.0f);
        Context.Output. Distance = SafeDistance;
        Context.Output. bCollisionAdjusted = true;

        UE_LOG(LogTemp, Verbose, TEXT("Stage6_CollisionResolve:  Basic collision at %. 2f, safe distance %.  2f"),
            Hit.Distance, SafeDistance);
    }
}

```

```
├── CreateSubSystems()

```

### 2. Stage 6执行流程

```
UCameraStage_CollisionResolve:: Execute()
├── 检查CollisionResolver是否可用
├── 如果可用：
│   ├── CollisionResolver->ResolveCollisions(Context)
│   └── CollisionResolver->Update(DeltaTime, Context)
└── 如果不可用：
    └── PerformBasicCollisionCheck(Context)  // Fallback
```
