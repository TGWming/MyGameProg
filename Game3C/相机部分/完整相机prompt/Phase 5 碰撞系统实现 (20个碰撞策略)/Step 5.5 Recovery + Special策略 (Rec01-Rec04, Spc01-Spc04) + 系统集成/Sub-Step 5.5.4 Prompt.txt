```cpp
// Copyright Epic Games, Inc.    All Rights Reserved.  

#pragma once

#include "CoreMinimal.h"
#include "CameraCollisionBase.h"
#include "CameraCollision_Special.generated.h"

/**
 * Special Strategy Declarations (SP01-SP04)
 * 
 * This file contains the 4 Special strategies for the collision system.  
 * Special strategies handle unique environmental situations that require
 * custom camera behavior beyond standard collision response.  
 * 
 * All special strategies inherit from UCameraCollision_SpecialBase which
 * provides common functionality for environment detection and response.  
 * 
 * Key Characteristics:
 * - Execute during the Special Phase of collision resolution
 * - Higher priority than standard collision strategies (150+)
 * - Handle specific environmental conditions
 * - May produce multiple adjustment types simultaneously
 * 
 * Special Types:
 * - SP01: TightSpace - Confined area handling (corridors, rooms)
 * - SP02: LowCeiling - Low overhead clearance handling
 * - SP03: CliffEdge - Edge/drop-off handling
 * - SP04: CornerCase - Corner geometry handling
 */


//========================================
// Special Base Class
//========================================

/**
 * UCameraCollision_SpecialBase
 * 
 * Abstract base class for all special strategies.   
 * Provides common functionality for environment detection and response.  
 */
UCLASS(Abstract, BlueprintType)
class MYPROJECT_API UCameraCollision_SpecialBase : public UCameraCollisionBase
{
    GENERATED_BODY()

public:
    UCameraCollision_SpecialBase();

    //========================================
    // Strategy Identity
    //========================================

    /** All special strategies belong to Special category */
    virtual ECollisionCategory GetStrategyCategory() const override { return ECollisionCategory::Special; }

    //========================================
    // Lifecycle
    //========================================

    /** Update special state each frame */
    virtual void Update(float DeltaTime, const FStageExecutionContext& Context) override;

    /** Reset special to initial state */
    virtual void Reset() override;

protected:
    //========================================
    // Common Special Methods
    //========================================

    /**
     * Perform environment probe in multiple directions
     * @param Context Execution context
     * @param ProbeDistance Distance to probe
     * @param OutHits Array of hit results for each direction
     * @return Number of directions blocked
     */
    int32 PerformEnvironmentProbe(
        const FStageExecutionContext& Context,
        float ProbeDistance,
        TArray<FHitResult>& OutHits
    ) const;

    /**
     * Check if a direction is blocked
     * @param Context Execution context
     * @param Direction Direction to check (normalized)
     * @param Distance Check distance
     * @param OutHit Hit result if blocked
     * @return True if direction is blocked
     */
    bool IsDirectionBlocked(
        const FStageExecutionContext& Context,
        const FVector& Direction,
        float Distance,
        FHitResult& OutHit
    ) const;

    /**
     * Calculate clearance in a direction
     * @param Context Execution context
     * @param Direction Direction to check
     * @param MaxDistance Maximum check distance
     * @return Clearance distance (MaxDistance if not blocked)
     */
    float GetClearanceInDirection(
        const FStageExecutionContext& Context,
        const FVector& Direction,
        float MaxDistance
    ) const;

    /**
     * Get camera location from context
     * @param Context Execution context
     * @return Camera world location
     */
    FVector GetCameraLocation(const FStageExecutionContext& Context) const;

    /**
     * Interpolate special factor for smooth transitions
     * @param Current Current factor value
     * @param Target Target factor value
     * @param DeltaTime Time since last frame
     * @param Speed Interpolation speed
     * @return Interpolated factor
     */
    float InterpolateSpecialFactor(float Current, float Target, float DeltaTime, float Speed) const;

    //========================================
    // Common Configuration
    //========================================

    /** Base detection distance for environment probes */
    UPROPERTY(EditAnywhere, BlueprintReadWrite, Category = "Special|Config", meta = (ClampMin = "50.0", ClampMax = "500.0"))
    float BaseProbeDistance;

    /** Transition speed for special mode activation */
    UPROPERTY(EditAnywhere, BlueprintReadWrite, Category = "Special|Config", meta = (ClampMin = "1.0", ClampMax = "20.0"))
    float TransitionSpeed;

    /** Minimum factor to consider special mode active */
    UPROPERTY(EditAnywhere, BlueprintReadWrite, Category = "Special|Config", meta = (ClampMin = "0.01", ClampMax = "0.5"))
    float ActivationThreshold;

    //========================================
    // Runtime State
    //========================================

    /** Current special mode factor (0-1) */
    UPROPERTY(BlueprintReadOnly, Category = "Special|State")
    float CurrentSpecialFactor;

    /** Is special mode currently active */
    UPROPERTY(BlueprintReadOnly, Category = "Special|State")
    bool bSpecialModeActive;

    /** Time in special mode */
    float TimeInSpecialMode;

    /** Cached delta time */
    float CachedDeltaTime;
};

/**
 * UCameraCollision_SP01_TightSpace
 * 
 * SP01: Handle confined/tight spaces
 * 
 * Detects when camera is in a confined area and adjusts behavior: 
 * - Increases FOV to maintain visibility
 * - Allows closer minimum distance
 * - Applies special handling for narrow corridors
 * 
 * Use Cases:
 * - Narrow corridors
 * - Small rooms
 * - Confined indoor spaces
 * 
 * Characteristics:
 * - Multi-directional environment probe
 * - FOV compensation
 * - Priority:  150
 */
UCLASS(BlueprintType, Blueprintable)
class MYPROJECT_API UCameraCollision_SP01_TightSpace : public UCameraCollision_SpecialBase
{
    GENERATED_BODY()

public:
    UCameraCollision_SP01_TightSpace();

    //========================================
    // Strategy Identity
    //========================================

    virtual ECollisionStrategyID GetStrategyType() const override 
    { 
        return ECollisionStrategyID:: Collision_SP01_TightSpace; 
    }
    
    virtual FString GetStrategyDescription() const override 
    { 
        return TEXT("Handle confined/tight spaces with FOV compensation"); 
    }
    
    virtual int32 GetPriority() const override { return 150; }

    //========================================
    // Execution
    //========================================

    virtual bool ShouldExecute(const FStageExecutionContext& Context) const override;

    virtual bool Execute(
        const FStageExecutionContext& Context,
        const FCollisionDetectionResult& DetectionResult,
        FCollisionResponseOutput& OutResponse
    ) override;

    virtual void Update(float DeltaTime, const FStageExecutionContext& Context) override;
    virtual void Reset() override;

    //========================================
    // Public Query
    //========================================

    /** Is currently in tight space */
    UFUNCTION(BlueprintPure, Category = "Special")
    bool IsInTightSpace() const { return bInTightSpace; }

    /** Get current tight space factor */
    UFUNCTION(BlueprintPure, Category = "Special")
    float GetTightSpaceFactor() const { return TightSpaceFactor; }

    // Runtime State

    /** Is in tight space */
```

### 1. 类继承结构

```
UCameraCollisionBase
└── UCameraCollision_SpecialBase (特殊环境基类)
    ├── UCameraCollision_SP01_TightSpace
    ├── UCameraCollision_SP02_LowCeiling
    ├── UCameraCollision_SP03_CliffEdge
    └── UCameraCollision_SP04_CornerCase
```

### 2. 优先级设计（高于其他类别）

| 策略 | 优先级 | 说明 |
|------|--------|------|
| SP04 CornerCase | 155 | 最高，角落优先处理 |
| SP01 TightSpace | 150 | 高，狭窄空间 |
| SP02 LowCeiling | 145 | 中高，低天花板 |
| SP03 CliffEdge | 140 | 中，悬崖边缘 |

### 3. SpecialBase提供的辅助方法

```cpp
PerformEnvironmentProbe()     // 多方向环境探测
IsDirectionBlocked()          // 检查方向是否被阻挡
GetClearanceInDirection()     // 获取某方向的净空距离
GetCameraLocation()           // 获取相机位置
InterpolateSpecialFactor()    // 平滑插值特殊因子
```

### 4. 四种Special策略特点

| 策略 | 检测方法 | 调整类型 | 适用场景 |
|------|----------|----------|----------|
| SP01 TightSpace | 多向探测 | FOV + 距离 | 走廊、小房间 |
| SP02 LowCeiling | 垂直探测 | Pitch + 位置 | 矮通道、洞穴 |
| SP03 CliffEdge | 地面探测 | 位置 + Pitch | 悬崖、平台边缘 |
| SP04 CornerCase | 角度探测 | 位置 + Yaw | 墙角、L型走廊 |
