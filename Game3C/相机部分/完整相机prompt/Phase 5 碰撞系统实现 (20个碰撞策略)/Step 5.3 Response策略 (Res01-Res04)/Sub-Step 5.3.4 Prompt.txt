**找到Res04的placeholder代码，替换为以下实现：**

```cpp
//========================================
// Res04:  Hybrid Implementation
//========================================

/**
 * Constructor - Configure hybrid response parameters
 * 
 * Hybrid response intelligently combines multiple response types:  
 * - Pull-in for distance adjustment
 * - Slide for lateral movement
 * - Rotation for angle adjustment
 * - Adaptive weights based on collision type
 */
UCameraCollision_Res04_Hybrid::UCameraCollision_Res04_Hybrid()
    : BasePullInWeight(0.6f)
    , BaseSlideWeight(0.3f)
    , BaseRotateWeight(0.1f)
    , MaxHybridSlide(50.0f)
    , MaxHybridRotation(10.0f)
    , HybridInterpSpeed(12.0f)
    , bAdaptiveWeights(true)
    , CurrentPullIn(0.0f)
    , TargetPullIn(0.0f)
    , CurrentSlide(FVector:: ZeroVector)
    , TargetSlide(FVector::ZeroVector)
    , CurrentRotation(FRotator::ZeroRotator)
    , TargetRotation(FRotator::ZeroRotator)
    , EffectivePullInWeight(0.6f)
    , EffectiveSlideWeight(0.3f)
    , EffectiveRotateWeight(0.1f)
{
    Priority = 120;  // Highest priority among response strategies
    BaseInterpSpeed = 12.0f;
    bCanBlend = true;
}

bool UCameraCollision_Res04_Hybrid::ShouldExecute(const FStageExecutionContext& Context) const
{
    return bIsEnabled;
}

bool UCameraCollision_Res04_Hybrid::Execute(
    const FStageExecutionContext& Context,
    const FCollisionDetectionResult& DetectionResult,
    FCollisionResponseOutput& OutResponse)
{
    // Check if response is needed
    if (!IsResponseNeeded(DetectionResult))
    {
        // No collision - smoothly return all adjustments to zero
        bool bStillActive = false;
        
        // Return pull-in
        if (FMath::Abs(CurrentPullIn) > 0.1f)
        {
            CurrentPullIn = InterpolateValue(CurrentPullIn, 0.0f, CachedDeltaTime, BaseInterpSpeed * 0.5f);
            bStillActive = true;
        }
        else
        {
            CurrentPullIn = 0.0f;
        }
        
        // Return slide
        if (! CurrentSlide.IsNearlyZero(0.5f))
        {
            CurrentSlide = InterpolateVector(CurrentSlide, FVector::ZeroVector, CachedDeltaTime, BaseInterpSpeed * 0.5f);
            bStillActive = true;
        }
        else
        {
            CurrentSlide = FVector::ZeroVector;
        }
        
        // Return rotation
        if (!CurrentRotation.IsNearlyZero(0.3f))
        {
            CurrentRotation = InterpolateRotator(CurrentRotation, FRotator:: ZeroRotator, CachedDeltaTime, BaseInterpSpeed * 0.3f);
            bStillActive = true;
        }
        else
        {
            CurrentRotation = FRotator:: ZeroRotator;
        }
        
        if (bStillActive)
        {
            // Output returning adjustments
            InitializeResponse(OutResponse);
            OutResponse.DistanceAdjustment = CurrentPullIn;
            OutResponse. PositionAdjustment = CurrentSlide;
            OutResponse. RotationAdjustment = CurrentRotation;
            OutResponse.ActiveDuration = ResponseTime;
            
            bResponseActive = true;
            SetActive(true);
            return true;
        }
        
        bResponseActive = false;
        SetActive(false);
        return false;
    }

    bResponseActive = true;
    SetActive(true);

    // Calculate adaptive weights based on collision type
    if (bAdaptiveWeights)
    {
        CalculateAdaptiveWeights(DetectionResult);
    }
    else
    {
        EffectivePullInWeight = BasePullInWeight;
        EffectiveSlideWeight = BaseSlideWeight;
        EffectiveRotateWeight = BaseRotateWeight;
    }

    // Calculate total adjustment needed
    float TotalAdjustmentNeeded = DetectionResult. DesiredDistance - DetectionResult.SafeDistance;
    float Urgency = DetectionResult.AdjustmentRatio;
    
    //========================================
    // Calculate Pull-In Component
    //========================================
    
    // Pull-in handles most of the distance adjustment
    TargetPullIn = -TotalAdjustmentNeeded * EffectivePullInWeight;
    
    //========================================
    // Calculate Slide Component
    //========================================
    
    // Slide helps when there's a clear lateral path
    FVector SlideDir = CalculateSlideDirection(
        DetectionResult. CollisionNormal,
        -Context.Output. Rotation.Vector()
    );
    
    float SlideAmount = TotalAdjustmentNeeded * EffectiveSlideWeight;
    SlideAmount = FMath::Min(SlideAmount, MaxHybridSlide);
    
    TargetSlide = SlideDir * SlideAmount;
    
    //========================================
    // Calculate Rotation Component
    //========================================
    
    // Rotation is used sparingly for fine adjustment
    FVector ToCollision = DetectionResult. HitResult.ImpactPoint - Context.Output.FocusPoint;
    ToCollision. Normalize();
    
    // Calculate rotation away from collision
    FVector CameraDir = -Context.Output. Rotation.Vector();
    FVector RotateAwayDir = FVector::CrossProduct(DetectionResult.CollisionNormal, FVector:: UpVector);
    
    if (RotateAwayDir.Dot(CameraDir) < 0.0f)
    {
        RotateAwayDir = -RotateAwayDir;
    }
    
    float RotateAmount = TotalAdjustmentNeeded * EffectiveRotateWeight;
    RotateAmount = FMath:: Min(RotateAmount, MaxHybridRotation);
    
    // Convert to rotation
    TargetRotation. Yaw = FMath::Sign(FVector:: CrossProduct(CameraDir, RotateAwayDir).Z) * RotateAmount;
    TargetRotation. Pitch = 0.0f;
    TargetRotation.Roll = 0.0f;
    
    //========================================
    // Apply Interpolation
    //========================================
    
    float EffectiveSpeed = HybridInterpSpeed * (1.0f + Urgency);
    
    CurrentPullIn = InterpolateValue(CurrentPullIn, TargetPullIn, CachedDeltaTime, EffectiveSpeed);
    CurrentSlide = InterpolateVector(CurrentSlide, TargetSlide, CachedDeltaTime, EffectiveSpeed);
    CurrentRotation = InterpolateRotator(CurrentRotation, TargetRotation, CachedDeltaTime, EffectiveSpeed * 0.7f);

    //========================================
    // Build Response Output
    //========================================
    
    InitializeResponse(OutResponse);
    OutResponse.DistanceAdjustment = CurrentPullIn;
    OutResponse. PositionAdjustment = CurrentSlide;
    OutResponse.RotationAdjustment = CurrentRotation;
    OutResponse.ActiveDuration = ResponseTime;

    UE_LOG(LogTemp, Verbose, TEXT("Res04 Hybrid: PullIn=%.1f, Slide=(%.1f,%.1f,%.1f), Rot=%.1f, Weights=(%.2f,%.2f,%.2f)"),
        CurrentPullIn,
        CurrentSlide.X, CurrentSlide.Y, CurrentSlide. Z,
        CurrentRotation.Yaw,
        EffectivePullInWeight, EffectiveSlideWeight, EffectiveRotateWeight);

    return true;
}

void UCameraCollision_Res04_Hybrid::Update(float DeltaTime, const FStageExecutionContext& Context)
{
    Super::Update(DeltaTime, Context);
}

void UCameraCollision_Res04_Hybrid::Reset()
{
    Super::Reset();
    
    CurrentPullIn = 0.0f;
    TargetPullIn = 0.0f;
    CurrentSlide = FVector::ZeroVector;
    TargetSlide = FVector::ZeroVector;
    CurrentRotation = FRotator::ZeroRotator;
    TargetRotation = FRotator:: ZeroRotator;
    
    EffectivePullInWeight = BasePullInWeight;
    EffectiveSlideWeight = BaseSlideWeight;
    EffectiveRotateWeight = BaseRotateWeight;
}

void UCameraCollision_Res04_Hybrid:: CalculateAdaptiveWeights(const FCollisionDetectionResult& DetectionResult)
{
    // Start with base weights
    EffectivePullInWeight = BasePullInWeight;
    EffectiveSlideWeight = BaseSlideWeight;
    EffectiveRotateWeight = BaseRotateWeight;
    
    // Analyze collision characteristics
    FVector CollisionNormal = DetectionResult.CollisionNormal;
    
    // Check if collision is from front (camera hitting wall behind character)
    float NormalDotUp = FMath:: Abs(CollisionNormal.Z);
    float NormalDotHorizontal = FMath::Sqrt(CollisionNormal.X * CollisionNormal.X + CollisionNormal. Y * CollisionNormal.Y);
    
    // Vertical surface (wall) - sliding is more effective
    if (NormalDotHorizontal > 0.7f)
    {
        EffectiveSlideWeight *= 1.5f;
        EffectivePullInWeight *= 0.8f;
    }
    // Horizontal surface (ceiling/floor) - pull-in is more effective
    else if (NormalDotUp > 0.7f)
    {
        EffectivePullInWeight *= 1.3f;
        EffectiveSlideWeight *= 0.5f;
        EffectiveRotateWeight *= 0.5f;
    }
    
    // In tight spaces, prefer pull-in over slide
    if (DetectionResult.bInTightSpace)
    {
        EffectivePullInWeight *= 1.5f;
        EffectiveSlideWeight *= 0.5f;
        EffectiveRotateWeight *= 0.3f;
    }
    
    // High urgency - emphasize pull-in for faster response
    if (DetectionResult.AdjustmentRatio > 0.7f)
    {
        EffectivePullInWeight *= 1.2f;
    }
    
    // Normalize weights to sum to 1.0
    float TotalWeight = EffectivePullInWeight + EffectiveSlideWeight + EffectiveRotateWeight;
    if (TotalWeight > 0.0f)
    {
        EffectivePullInWeight /= TotalWeight;
        EffectiveSlideWeight /= TotalWeight;
        EffectiveRotateWeight /= TotalWeight;
    }
}

FVector UCameraCollision_Res04_Hybrid:: CalculateSlideDirection(
    const FVector& CollisionNormal,
    const FVector& CameraDirection) const
{
    // Project camera direction onto collision plane
    FVector SlideDir = FVector::VectorPlaneProject(CameraDirection, CollisionNormal);
    
    if (SlideDir. IsNearlyZero())
    {
        // Camera direction is perpendicular to surface
        FVector Right = FVector::CrossProduct(CollisionNormal, FVector:: UpVector);
        if (Right.IsNearlyZero())
        {
            Right = FVector::CrossProduct(CollisionNormal, FVector:: ForwardVector);
        }
        SlideDir = Right. GetSafeNormal();
    }
    else
    {
        SlideDir. Normalize();
    }
    
    // Prefer horizontal sliding
    SlideDir. Z *= 0.3f;
    SlideDir. Normalize();
    
    return SlideDir;
}
```

【代码设计说明】

### 1. Res04 Hybrid 自适应权重算法

```
CalculateAdaptiveWeights() 分析碰撞特征：

碰撞面类型判断：
├── 垂直面（墙壁）：NormalDotHorizontal > 0.7
│   ├── SlideWeight × 1.5
│   └── PullInWeight × 0.8
│
├── 水平面（天花板/地面）：NormalDotUp > 0.7
│   ├── PullInWeight × 1.3
│   ├── SlideWeight × 0.5
│   └── RotateWeight × 0.5
│
└── 特殊情况：
    ├── 狭窄空间：强调PullIn
    └── 高紧急度：强调PullIn

最后归一化：确保权重总和为1.0
```

### 2. 三组件混合

```
Execute() 计算流程：
├── 1. 计算自适应权重
├── 2. PullIn分量：-TotalAdjustment × PullInWeight
├── 3. Slide分量：SlideDir × (Adjustment × SlideWeight)
├── 4. Rotate分量：基于碰撞法线的旋转
├── 5. 分别插值三个分量
└── 6. 合并输出

恢复流程：
├── 三个分量以不同速度恢复
├── PullIn/Slide：0.5x速度
└── Rotate：0.3x速度（最慢，避免眩晕）
```

### 3. 四种Response策略总结

| 策略 | 优先级 | 调整内容 | 最佳场景 |
|------|--------|----------|----------|
| Res01 PullIn | 100 | 距离 | 简单、默认 |
| Res02 Slide | 110 | 位置 | 墙壁、平面 |
| Res03 Rotate | 105 | 旋转 | 复杂几何 |
| Res04 Hybrid | 120 | 全部 | 智能适应 |
