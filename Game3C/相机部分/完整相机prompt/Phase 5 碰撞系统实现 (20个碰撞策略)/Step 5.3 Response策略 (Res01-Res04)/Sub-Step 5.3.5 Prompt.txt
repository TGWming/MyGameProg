//========================================
// Res04: Hybrid
//========================================

UCameraCollision_Res04_Hybrid::UCameraCollision_Res04_Hybrid()
    : PullInWeight(0.7f)
    , MaxHybridSlide(50.0f)
    , ResponseSpeed(400.0f)
    , CurrentPullIn(0.0f)
    , CurrentSlide(FVector::ZeroVector)
{
}

bool UCameraCollision_Res04_Hybrid::ShouldExecute(const FStageExecutionContext& Context, const FCameraStateConfig& StateConfig) const
{
    // Execute when collision is detected and hybrid response is preferred
    return bIsEnabled && StateConfig.Collision.Response.bEnableHybrid;
}

bool UCameraCollision_Res04_Hybrid::Execute(
    const FStageExecutionContext& Context,
    const FCameraStateConfig& StateConfig,
    const FCollisionDetectionResult& DetectionResult,
    FCollisionResponseOutput& OutResponse)
{
    if (!DetectionResult. bCollisionDetected)
    {
        // Decay both adjustments when no collision
        bool bStillActive = false;

        if (FMath::Abs(CurrentPullIn) > 0.1f)
        {
            CurrentPullIn = FMath::FInterpTo(CurrentPullIn, 0.0f, Context.DeltaTime, ResponseSpeed / 100.0f);
            bStillActive = true;
        }
        else
        {
            CurrentPullIn = 0.0f;
        }

        if (! CurrentSlide.IsNearlyZero())
        {
            float DecaySpeed = ResponseSpeed;
            FVector DecayAmount = CurrentSlide.GetSafeNormal() * DecaySpeed * Context.DeltaTime;
            
            if (DecayAmount.Size() >= CurrentSlide.Size())
            {
                CurrentSlide = FVector:: ZeroVector;
            }
            else
            {
                CurrentSlide -= DecayAmount;
                bStillActive = true;
            }
        }

        if (bStillActive)
        {
            OutResponse.StrategyType = GetStrategyType();
            OutResponse. DistanceAdjustment = -CurrentPullIn;
            OutResponse. PositionAdjustment = CurrentSlide;
            OutResponse. Weight = 0.5f;
            OutResponse.Priority = GetPriority();
        }

        bIsActive = bStillActive;
        return bStillActive;
    }

    bIsActive = true;

    // Calculate total adjustment needed
    float TotalAdjustment = DetectionResult.AdjustmentAmount;

    // Split between pull-in and slide based on weight
    float TargetPullIn = TotalAdjustment * PullInWeight;
    float SlideAmount = TotalAdjustment * (1.0f - PullInWeight);
    SlideAmount = FMath::Min(SlideAmount, MaxHybridSlide);

    // Calculate slide direction
    FVector CollisionNormal = DetectionResult. CollisionNormal;
    FVector CameraDirection = -Context. Output.Rotation.Vector();

    // Prefer horizontal slide
    FVector SlideDirection = FVector::CrossProduct(CollisionNormal, FVector:: UpVector);
    if (SlideDirection.IsNearlyZero())
    {
        SlideDirection = FVector::CrossProduct(CollisionNormal, CameraDirection);
    }
    SlideDirection.Normalize();

    // Determine slide direction consistency (prefer same direction as previous frame)
    if (FVector::DotProduct(SlideDirection, CurrentSlide) < 0.0f && ! CurrentSlide.IsNearlyZero())
    {
        SlideDirection = -SlideDirection;
    }

    FVector TargetSlide = SlideDirection * SlideAmount;

    // Validate slide position
    FVector FocusPoint = Context.Output. FocusPoint;
    FVector TestPosition = DetectionResult. SafeLocation + TargetSlide;

    FCollisionQueryParams Params;
    Params.AddIgnoredActor(Context.InputContext.CharacterActor.Get());

    FHitResult TestHit;
    bool bSlideBlocked = LineTrace(FocusPoint, TestPosition, TestHit, Params);

    if (bSlideBlocked)
    {
        // Reduce slide, increase pull-in
        TargetSlide *= 0.5f;
        TargetPullIn += SlideAmount * 0.25f;
    }

    // Interpolate to targets
    float MaxChangeThisFrame = ResponseSpeed * Context.DeltaTime;

    // Pull-in interpolation
    float PullInDelta = TargetPullIn - CurrentPullIn;
    if (FMath::Abs(PullInDelta) > MaxChangeThisFrame)
    {
        CurrentPullIn += FMath::Sign(PullInDelta) * MaxChangeThisFrame;
    }
    else
    {
        CurrentPullIn = TargetPullIn;
    }

    // Slide interpolation
    FVector SlideDelta = TargetSlide - CurrentSlide;
    if (SlideDelta.Size() > MaxChangeThisFrame)
    {
        CurrentSlide += SlideDelta.GetSafeNormal() * MaxChangeThisFrame;
    }
    else
    {
        CurrentSlide = TargetSlide;
    }

    // Build response
    OutResponse.StrategyType = GetStrategyType();
    OutResponse.DistanceAdjustment = -CurrentPullIn;
    OutResponse.PositionAdjustment = CurrentSlide;
    OutResponse.Weight = 1.0f;
    OutResponse.Priority = GetPriority();
    OutResponse.bOverride = false;

    UE_LOG(LogTemp, Verbose, TEXT("Res04 Hybrid: PullIn:  %.2f, Slide: (%.2f, %.2f, %.2f)"),
        CurrentPullIn, CurrentSlide. X, CurrentSlide.Y, CurrentSlide.Z);

    return true;
}
