```cpp
//========================================
// Det04: Capsule Trace Implementation
//========================================

/**
 * Constructor - Configure capsule trace detection parameters
 * 
 * Capsule trace is best for vertical clearance:  
 * - Better matches the "volume" the camera occupies
 * - Good for tight corridors and low ceilings
 * - Can be oriented along camera direction or vertical
 */
UCameraCollision_Det04_CapsuleTrace::  UCameraCollision_Det04_CapsuleTrace()
    : CapsuleRadius(12.0f)
    , CapsuleHalfHeight(30.0f)
    , bOrientAlongDirection(true)
    , bVerticalCapsule(false)
{
    // Medium-high priority
    Priority = 115;
    SafetyMargin = 10.0f;
}

bool UCameraCollision_Det04_CapsuleTrace:: Execute(
    const FStageExecutionContext& Context,
    const FCollisionDetectionResult& DetectionResult,
    FCollisionResponseOutput& OutResponse)
{
    // Calculate trace endpoints
    FVector TraceStart, TraceEnd;
    CalculateTraceEndpoints(Context, TraceStart, TraceEnd);
    
    FVector TraceDirection = (TraceEnd - TraceStart).GetSafeNormal();
    float TraceDistance = (TraceEnd - TraceStart).Size();
    
    // Determine capsule orientation
    FQuat CapsuleRotation;
    
    if (bVerticalCapsule)
    {
        // Vertical capsule (standing upright)
        CapsuleRotation = FQuat::Identity;
    }
    else if (bOrientAlongDirection)
    {
        // Orient capsule along trace direction
        CapsuleRotation = FQuat::FindBetweenNormals(FVector:: UpVector, TraceDirection);
    }
    else
    {
        // Use camera rotation
        CapsuleRotation = Context.Output. Rotation. Quaternion();
    }
    
    // Setup collision shape
    FCollisionShape CapsuleShape;
    CapsuleShape.SetCapsule(CapsuleRadius, CapsuleHalfHeight);
    
    // Setup trace parameters
    FCollisionQueryParams Params = GetDefaultTraceParams();
    
    // Perform capsule sweep
    UWorld* World = GetWorld();
    if (!World)
    {
        SetActive(false);
        return false;
    }
    
    FHitResult Hit;
    bool bHit = World->SweepSingleByChannel(
        Hit,
        TraceStart,
        TraceEnd,
        CapsuleRotation,
        TraceChannel,
        CapsuleShape,
        Params
    );
    
    // Debug visualization
#if ENABLE_DRAW_DEBUG
    if (bDrawDebugTraces)
    {
        FColor TraceColor = bHit ? FColor::Red : FColor::Green;
        
        // Draw start capsule
        DrawDebugCapsule(World, TraceStart, CapsuleHalfHeight, CapsuleRadius, 
            CapsuleRotation, FColor:: Cyan, false, 0.0f, 0, 0.5f);
        
        // Draw end capsule
        DrawDebugCapsule(World, TraceEnd, CapsuleHalfHeight, CapsuleRadius, 
            CapsuleRotation, TraceColor, false, 0.0f, 0, 0.5f);
        
        // Draw trace line
        DrawDebugLine(World, TraceStart, TraceEnd, TraceColor, false, 0.0f, 0, 1.0f);
        
        if (bHit)
        {
            // Draw hit location
            DrawDebugCapsule(World, Hit.Location, CapsuleHalfHeight, CapsuleRadius,
                CapsuleRotation, FColor::Red, false, 0.0f, 0, 1.0f);
            
            // Draw impact normal
            DrawDebugDirectionalArrow(World, Hit.ImpactPoint,
                Hit.ImpactPoint + Hit.ImpactNormal * 40.0f,
                15.0f, FColor::Blue, false, 0.0f, 0, 1.0f);
        }
    }
#endif
    
    if (bHit)
    {
        // Modify detection result
        FCollisionDetectionResult& MutableResult = const_cast<FCollisionDetectionResult&>(DetectionResult);
        
        // Fill basic hit info
        MutableResult.bCollisionDetected = true;
        MutableResult.HitResult = Hit;
        MutableResult. CollisionNormal = Hit.ImpactNormal;
        MutableResult.CollisionDistance = Hit.Distance;
        
        // Calculate safe distance (account for capsule size)
        // Use the larger of radius or half-height for safety margin
        float CapsuleSafetyOffset = FMath::Max(CapsuleRadius, CapsuleHalfHeight * 0.5f);
        float TotalSafetyOffset = CapsuleSafetyOffset + SafetyMargin;
        MutableResult.SafeDistance = FMath::Max(MinDistance, Hit.Distance - TotalSafetyOffset);
        
        // Calculate safe location
        MutableResult.SafeLocation = TraceStart + TraceDirection * MutableResult.SafeDistance;
        
        // Store desired values
        MutableResult. DesiredDistance = Context.Output.Distance;
        MutableResult.DesiredLocation = TraceEnd;
        
        // Calculate adjustment ratio
        float TotalAdjustableDistance = Context.Output.Distance - MinDistance;
        if (TotalAdjustableDistance > 0.0f)
        {
            float AdjustmentNeeded = Context.Output.Distance - MutableResult.SafeDistance;
            MutableResult.AdjustmentRatio = FMath::Clamp(AdjustmentNeeded / TotalAdjustableDistance, 0.0f, 1.0f);
        }
        
        // Check for tight space (capsule significantly compressed)
        if (MutableResult.AdjustmentRatio > 0.7f)
        {
            MutableResult.bInTightSpace = true;
        }
        
        UE_LOG(LogTemp, Verbose, TEXT("Det04 CapsuleTrace:  Hit at %.1f (r=%.1f, h=%.1f), safe=%.1f, tight=%d"),
            Hit.Distance, CapsuleRadius, CapsuleHalfHeight, 
            MutableResult.SafeDistance, MutableResult.bInTightSpace ?  1 : 0);
        
        SetActive(true);
        return true;
    }
    
    SetActive(false);
    return false;
}
```

【代码设计说明】

### 1. Det04 CapsuleTrace 特性

```
胶囊体方向选项：
├── bVerticalCapsule = true：垂直站立（检测低天花板）
├── bOrientAlongDirection = true：沿射线方向（默认）
└── 否则：使用相机旋转方向

安全距离计算：
├── CapsuleSafetyOffset = Max(Radius, HalfHeight * 0.5)
├── TotalSafetyOffset = CapsuleSafetyOffset + SafetyMargin
└── SafeDistance = Max(MinDistance, HitDistance - TotalSafetyOffset)

狭窄空间检测：
└── if (AdjustmentRatio > 0.7) → bInTightSpace = true
```

### 2. 四种Detection策略对比

| 策略 | 形状 | 优先级 | 最佳场景 | 性能 |
|------|------|--------|----------|------|
| Det01 LineTrace | 射线 | 100 | 开阔区域、备用 | 最快 |
| Det02 SphereTrace | 球体 | 120 | 通用、默认 | 快 |
| Det03 MultiRayTrace | 多射线 | 110 | 复杂环境 | 中等 |
| Det04 CapsuleTrace | 胶囊 | 115 | 走廊、低顶 | 中等 |

### 3. 注册顺序

```
CreateAllStrategies()
├── Det01 LineTrace     (Priority:  100)
├── Det02 SphereTrace   (Priority: 120) ← 最高优先级
├── Det03 MultiRayTrace (Priority: 110)
└── Det04 CapsuleTrace  (Priority: 115)

执行时按优先级降序：Det02 → Det04 → Det03 → Det01
```
