//========================================
// Det04: Predictive
//========================================

UCameraCollision_Det04_Predictive:: UCameraCollision_Det04_Predictive()
    : PredictionTime(0.2f)
    , MinSpeedForPrediction(50.0f)
    , PreviousCameraPosition(FVector::ZeroVector)
    , PreviousFocusPoint(FVector::ZeroVector)
    , CameraVelocity(FVector::ZeroVector)
    , bHasPreviousData(false)
{
}

bool UCameraCollision_Det04_Predictive::Execute(
    const FStageExecutionContext& Context,
    const FCameraStateConfig& StateConfig,
    const FCollisionDetectionResult& DetectionResult,
    FCollisionResponseOutput& OutResponse)
{
    FCollisionDetectionResult& MutableResult = const_cast<FCollisionDetectionResult&>(DetectionResult);

    // Calculate current camera position
    FVector FocusPoint = Context.Output. FocusPoint;
    FVector CameraDirection = -Context. Output.Rotation.Vector();
    float TargetDistance = Context.Output.Distance;
    FVector CurrentCameraPosition = FocusPoint + CameraDirection * TargetDistance;

    // Calculate camera velocity
    if (bHasPreviousData && Context.DeltaTime > 0.0f)
    {
        CameraVelocity = (CurrentCameraPosition - PreviousCameraPosition) / Context.DeltaTime;
    }
    else
    {
        CameraVelocity = FVector:: ZeroVector;
    }

    // Store for next frame
    PreviousCameraPosition = CurrentCameraPosition;
    PreviousFocusPoint = FocusPoint;
    bHasPreviousData = true;

    // Check if moving fast enough for prediction
    float Speed = CameraVelocity.Size();
    if (Speed < MinSpeedForPrediction)
    {
        bIsActive = false;
        return false;
    }

    // Calculate predicted position
    FVector PredictedCameraPosition = CurrentCameraPosition + CameraVelocity * PredictionTime;
    FVector PredictedFocusPoint = FocusPoint + (Context.InputContext.CharacterVelocity * PredictionTime);

    // Setup trace
    FCollisionQueryParams Params;
    Params. bTraceComplex = false;
    Params.AddIgnoredActor(Context.InputContext.CharacterActor.Get());
    if (Context.InputContext.TargetActor.IsValid())
    {
        Params.AddIgnoredActor(Context.InputContext.TargetActor.Get());
    }

    // Check for collision along predicted path
    FHitResult Hit;
    bool bHit = LineTrace(CurrentCameraPosition, PredictedCameraPosition, Hit, Params);

    if (!bHit)
    {
        // Also check from predicted focus to predicted camera
        bHit = LineTrace(PredictedFocusPoint, PredictedCameraPosition, Hit, Params);
    }

    if (bHit)
    {
        // Collision predicted - calculate safe distance
        MutableResult.bCollisionDetected = true;
        MutableResult.HitResult = Hit;
        MutableResult.CollisionNormal = Hit.ImpactNormal;

        // Calculate time until collision
        float DistanceToHit = FVector:: Dist(CurrentCameraPosition, Hit.ImpactPoint);
        float TimeToCollision = (Speed > 0.0f) ? (DistanceToHit / Speed) : PredictionTime;

        // Calculate safe distance based on how soon collision will occur
        // Closer collision = more aggressive response
        float Urgency = 1.0f - FMath::Clamp(TimeToCollision / PredictionTime, 0.0f, 1.0f);

        float SafeOffset = StateConfig.Collision.Detection.CollisionPadding * (1.0f + Urgency);
        MutableResult.SafeDistance = FMath::Max(
            TargetDistance - (MutableResult.AdjustmentAmount * Urgency),
            StateConfig. Collision.Detection.MinCameraDistance
        );

        // Safe location
        MutableResult.SafeLocation = FocusPoint + CameraDirection * MutableResult.SafeDistance;

        // Adjustment scaled by urgency
        MutableResult.AdjustmentAmount = (TargetDistance - MutableResult. SafeDistance) * Urgency;

        UE_LOG(LogTemp, Verbose, TEXT("Det04 Predictive:  Collision predicted in %.2fs, urgency %.2f"),
            TimeToCollision, Urgency);

        bIsActive = true;
        return true;
    }

    bIsActive = false;
    return false;
}
