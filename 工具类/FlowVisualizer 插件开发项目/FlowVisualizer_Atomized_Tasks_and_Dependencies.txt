    struct FFlowEvent
      double Timestamp;          // FPlatformTime::Seconds()
      uint64 FrameNumber;        // GFrameNumber

      TArray<FFlowNode> Nodes;

    所有结构体添加 FLOWVISUALIZERRUNTIME_API 导出宏

Task P2.2 — 实现 FFlowTracer 单例
──────────────────────────────────────
  依赖: P2.1
  
  输入: FlowPipelineTypes.h (FFlowEvent)
  
  输出文件:
    FlowVisualizer/Source/FlowVisualizerRuntime/Public/FlowTracer.h
    FlowVisualizer/Source/FlowVisualizerRuntime/Private/FlowTracer.cpp
  
  详细要求:
    class FFlowTracer:
      static FFlowTracer& Get();          // Singleton (Meyer's singleton)
      
      void Signal(FName NodeID, const FString& Data = TEXT(""));
        - 构造 FFlowEvent (Timestamp, FrameNumber, NodeID, Data)
        - 加锁写入 EventBuffer
        - 广播 OnFlowEvent Delegate
      
      TArray<FFlowEvent> GetRecentEvents(int32 Count) const;
        - 从 EventBuffer 中取最近 Count 条
        - 加锁读取
      
      void ClearBuffer();
        - 清空 EventBuffer
        - 加锁
      
      DECLARE_MULTICAST_DELEGATE_OneParam(FOnFlowEvent, const FFlowEvent&);
      FOnFlowEvent OnFlowEvent;
    
    Private:
      TArray<FFlowEvent> EventBuffer;     // 用 TArray 模拟环形缓冲
      int32 BufferHead = 0;               // 写入位置
      int32 BufferCapacity = 1024;        // 容量
      FCriticalSection BufferLock;         // 防御式线程安全
    
    注意:
      - UE4.27 中 TCircularBuffer 可能签名不同，使用 TArray + Head/Tail 更安全
      - Signal() 中 OnFlowEvent.Broadcast() 在锁外调用（避免死锁）
  
  验证标准:
    - Get() 返回同一实例
    - Signal() 不崩溃
    - GetRecentEvents() 返回正确数量
    - ClearBuffer() 后 GetRecentEvents() 返回空


Task P2.3 — 实现 FLOW_SIGNAL 宏
──────────────────────────────────────
  依赖: P2.2
  
  输入: FlowTracer.h
  
  输出文件:
    FlowVisualizer/Source/FlowVisualizerRuntime/Public/FlowTracerMacros.h
  
  详细要求:
    #pragma once
    #include "FlowTracer.h"
    
    #if !UE_BUILD_SHIPPING
    
    #define FLOW_SIGNAL(NodeID, Data) \
        FFlowTracer::Get().Signal(FName(TEXT(NodeID)), Data)
    
    #define FLOW_SIGNAL_SIMPLE(NodeID) \
        FFlowTracer::Get().Signal(FName(TEXT(NodeID)))
    
    #define FLOW_SIGNAL_FMT(NodeID, Fmt, ...) \
        FFlowTracer::Get().Signal(FName(TEXT(NodeID)), \
            FString::Printf(TEXT(Fmt), ##__VA_ARGS__))
    
    #else
    
    #define FLOW_SIGNAL(NodeID, Data)
    #define FLOW_SIGNAL_SIMPLE(NodeID)
    #define FLOW_SIGNAL_FMT(NodeID, Fmt, ...)
    
    #endif
  
  验证标准:
    - 在任意 .cpp 文件中 #include "FlowTracerMacros.h" 后
      调用 FLOW_SIGNAL("Test", TEXT("hello")) 编译通过
    - Shipping 构建时宏展开为空

Task P5.1 — 实现 DrawNode 绘制节点文字
──────────────────────────────────────
  依赖: P4.2 (需要 CurrentPipelineID 和管线数据)
  
  输入: 当前选中的 FFlowPipeline
  
  输出文件:
    FlowVisualizer/Source/FlowVisualizerEditor/Private/SFlowVisualizerPanel.cpp
    (修改 OnPaint)
  
  详细要求:
    遍历当前 Pipeline 的所有 Nodes:
      - 将 RelativePosition (0~1) 转换为窗口像素坐标
        PixelX = RelativePosition.X * (CanvasWidth - Margin*2) + Margin
        PixelY = RelativePosition.Y * (CanvasHeight - TopBar - BottomBar) + TopBar
      
      - 绘制节点名文字:
        用 FSlateDrawElement::MakeText
        字体: Roboto Bold 14pt
        颜色: 暗灰 #666666 (默认非活跃)
        格式: "[{DisplayName}]"
      
      - 节点名两侧加方括号 [] 以模拟终端风格
      
      - (占位) 节点下方预留数据文字行的空间
  
  验证标准:
    - 选择 Demo Pipeline 后窗口中显示 [Input] [Process] [Output] 三个文字
    - 位置与 RelativePosition 一致


Task P5.2 — 实现 DrawEdge 绘制连线箭头

        三角形尺寸: 8x6 像素
        颜色: 同连线
      
      - 对于垂直连线（节点上下排列）:
        绘制 L 形或直线折线

Task P7.2 — DrawFlowDot 流动光点绘制

    For each Edge:
      Elapsed = CurrentTime - EdgeLastActiveTime[Edge]
      
      if Elapsed < HighlightFadeDuration:
          // 1. 连线颜色变亮
          EdgeAlpha = 1.0 - (Elapsed / HighlightFadeDuration)
          LineColor = Lerp(EdgeDefault, EdgeActive, EdgeAlpha)
          重绘连线用 LineColor
          
          // 2. 计算光点位置
          SourcePos = SourceNode pixel position
          TargetPos = TargetNode pixel position
          EdgeLength = Distance(SourcePos, TargetPos)

              Progress = FMath::Fmod(Elapsed * FlowDotSpeed / EdgeLength, 1.0f)
              DotPos = FMath::Lerp(SourcePos, TargetPos, Progress)
              DotAlpha = EdgeAlpha
              
              // 3. 绘制光点
              FSlateDrawElement::MakeBox(
                  OutDrawElements,
                  LayerId + 2,
                  Geometry.ToPaintGeometry(DotPos - FVector2D(3,3), FVector2D(6,6)),
                  WhiteBrush,
                  ESlateDrawEffect::None,
                  FLinearColor(1,1,1, DotAlpha)
              )
      else:
          // 连线保持默认灰色，无光点
