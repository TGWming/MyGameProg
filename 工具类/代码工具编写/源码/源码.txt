### 4. 粘性目标切换（Sticky Feeling）

```cpp
bool UTargetSystemComponent::ShouldSwitchTargetActor(const float AxisValue)
{
    if (bEnableStickyTarget)
    {
        // 累积输入值（避免轻微摇杆移动就切换）
        StartRotatingStack += (AxisValue != 0) ? AxisValue * AxisMultiplier 
            : (StartRotatingStack > 0 ? -AxisMultiplier : AxisMultiplier);
        
        // 未达到阈值，不切换
        if (FMath::Abs(StartRotatingStack) < StickyRotationThreshold)
        {
            bDesireToSwitch = false;
            return false;
        }
        
        // 达到阈值，允许切换
        if (StartRotatingStack * AxisValue > 0)
        {
            StartRotatingStack = StartRotatingStack > 0 
                ? StickyRotationThreshold 
                : -StickyRotationThreshold;
        }
        
        bDesireToSwitch = true;
        return true;
    }
    
    // 非粘性模式：直接检查阈值
    return FMath::Abs(AxisValue) > StartRotatingThreshold;
}
```

**设计亮点**：防止手柄轻微漂移导致频繁切换！

```cpp
bool UTargetSystemComponent::ShouldBreakLineOfSight() const
{
    // 射线检测是否被遮挡
    TArray<AActor*> ActorsToIgnore = GetAllActorsOfClass(TargetableActors);
    ActorsToIgnore.Remove(LockedOnTargetActor);
    
    FHitResult HitResult;
    const bool bHit = LineTrace(HitResult, LockedOnTargetActor, ActorsToIgnore);
    
    // 如果射线击中了其他物体（而非目标），说明视线被遮挡
    if (bHit && HitResult.GetActor() != LockedOnTargetActor)
    {
        return true;
    }
    
    return false;
}

// 延迟断开（避免短暂遮挡就解锁）
```

### 2. 软锁定系统（Soft Lock）- 独特创新

```cpp
void UDSLockArmComponent::TickComponent(float DeltaTime, ...)
{
    if (IsCameraLockedToTarget())
    {
        // 硬锁定模式：持续跟踪目标
        
        // 检查距离，太远自动断开
        float Distance = (CameraTarget->GetComponentLocation() - GetComponentLocation()).Size();
        if (Distance > MaxTargetLockDistance + CameraTarget->GetScaledSphereRadius())
        {
            if (bUseSoftLock)
            {
                // 软锁定：尝试切换到范围内的新目标
                if (UDSTargetComponent* NewTarget = GetLockTarget())
                    LockToTarget(NewTarget);
                else
                    BreakTargetLock();
            }
            else
            {
                // 硬锁定：直接断开
                BreakTargetLock();
            }
        }
    }
    else if (bUseSoftLock)  // 无锁定但开启软锁定
    {
        // 🔑 自动锁定进入范围的敌人！
        if (UDSTargetComponent* NewTarget = GetLockTarget())
        {
            if (!bSoftlockRequiresReset)  // 需要重置标志
                LockToTarget(NewTarget);
        }
        else
        {
            // 没有目标时重置软锁定
            bSoftlockRequiresReset = false;
        }
    }
}
```

### 3. 智能目标切换系统

```cpp
void UDSLockArmComponent::SwitchTarget(EDirection SwitchDirection)
{
    TArray<UDSTargetComponent*> AvailableTargets = GetTargetComponents();
    if (AvailableTargets.Num() < 2) return;
    
    FVector CurrentTargetDir = (CameraTarget->GetComponentLocation() - GetComponentLocation()).GetSafeNormal();
    TArray<UDSTargetComponent*> ViableTargets;
    
    for (UDSTargetComponent* Target : AvailableTargets)
    {
        if (Target == CameraTarget) continue;
        
        FVector TargetDir = (Target->GetComponentLocation() - GetComponentLocation()).GetSafeNormal();
        
        // 🔑 使用叉积判断左右方向！
        FVector Cross = FVector::CrossProduct(CurrentTargetDir, TargetDir);
        
        if ((SwitchDirection == EDirection::Left && Cross.Z < 0.f)   // Z负=左侧
            || (SwitchDirection == EDirection::Right && Cross.Z > 0.f))  // Z正=右侧
        {
            ViableTargets.AddUnique(Target);
        }
    }
    
    // 选择角度最小的目标
    int32 BestDotIdx = 0;
    for (int32 i = 1; i < ViableTargets.Num(); i++)
    {
        float BestDot = FVector::DotProduct(CurrentTargetDir, 
            (ViableTargets[BestDotIdx]->GetComponentLocation() - GetComponentLocation()).GetSafeNormal());
        float TestDot = FVector::DotProduct(CurrentTargetDir, 
            (ViableTargets[i]->GetComponentLocation() - GetComponentLocation()).GetSafeNormal());
        
        if (TestDot > BestDot)  // 点积越大，角度越小
            BestDotIdx = i;
    }
    
    LockToTarget(ViableTargets[BestDotIdx]);
}
```

**切换逻辑精妙之处**：
1. 使用叉积Z分量判断左右（3D数学的巧妙应用）
2. 使用点积选择最近角度的目标
3. 防止频繁切换的延迟机制
